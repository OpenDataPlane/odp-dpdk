116static void clear_sched_queues(test_globals_t *globals)
117{
118 odp_event_t ev;
119 odp_buffer_t buf;
120 test_event_t *event;
121 int i, j;
122 odp_queue_t fromq;
123
124 /* Allocate the cool_down event. */
125 buf = odp_buffer_alloc(globals->pool);
126 if (buf == ODP_BUFFER_INVALID)
127 ODPH_ABORT("Buffer alloc failed.\n");
128
129 event = odp_buffer_addr(buf);
130 event->type = COOL_DOWN;
131 ev = odp_buffer_to_event(buf);
132
133 for (i = 0; i < NUM_PRIOS; i++) {
134 for (j = 0; j < globals->args.prio[i].queues; j++) {
135 /* Enqueue cool_down event on each queue. */
136 if (odp_queue_enq(globals->queue[i][j], ev))
137 ODPH_ABORT("Queue enqueue failed.\n");
138
139 /* Invoke scheduler until cool_down event has been
140 * received. */
141 while (1) {
142 ev = odp_schedule(NULL, ODP_SCHED_WAIT);
143 buf = odp_buffer_from_event(ev);
144 event = odp_buffer_addr(buf);
145 if (event->type == COOL_DOWN)
146 break;
147 odp_event_free(ev);
148 }
149 }
150 }
151
152 /* Free the cool_down event. */
153 odp_event_free(ev);
154
155 /* Call odp_schedule() to trigger a release of any scheduler context. */
156 ev = odp_schedule(&fromq, ODP_SCHED_NO_WAIT);
157 if (ev != ODP_EVENT_INVALID)
158 ODPH_ABORT("Queue %" PRIu64 " not empty.\n",
159 odp_queue_to_u64(fromq));
160}

176static int enqueue_events(int prio, int num_queues, int num_events,
177 int num_samples, odp_bool_t div_events,
178 test_globals_t *globals)
179{
180 odp_buffer_t buf[num_events + num_samples];
181 odp_event_t ev[num_events + num_samples];
182 odp_queue_t queue;
183 test_event_t *event;
184 int i, j, ret;
185 int enq_events;
186 int events_per_queue;
187 int tot_events;
188 int rdy_events = 0;
189
190 tot_events = num_events + num_samples;
191
192 if (!num_queues || !tot_events)
193 return 0;
194
195 events_per_queue = tot_events;
196 if (div_events)
197 events_per_queue = (tot_events + num_queues - 1) / num_queues;
198
199 for (i = 0; i < num_queues; i++) {
200 queue = globals->queue[prio][i];
201
202 ret = odp_buffer_alloc_multi(globals->pool, buf,
203 events_per_queue);
204 if (ret != events_per_queue) {
205 ODPH_ERR("Buffer alloc failed. Try increasing EVENT_POOL_SIZE.\n");
206 ret = ret < 0 ? 0 : ret;
207 odp_buffer_free_multi(buf, ret);
208 return -1;
209 }
210 for (j = 0; j < events_per_queue; j++) {
211 if (!odp_buffer_is_valid(buf[j])) {
212 ODPH_ERR("Buffer alloc failed\n");
213 odp_buffer_free_multi(buf, events_per_queue);
214 return -1;
215 }
216
217 event = odp_buffer_addr(buf[j]);
218 memset(event, 0, sizeof(test_event_t));
219
220 /* Latency isn't measured from the first processing
221 * rounds. */
222 if (num_samples > 0) {
223 event->type = WARM_UP;
224 event->warm_up_rounds = 0;
225 num_samples--;
226 } else {
227 event->type = TRAFFIC;
228 }
229 event->src_idx[prio] = i;
230 event->prio = prio;
231 ev[j] = odp_buffer_to_event(buf[j]);
232 }
233
234 enq_events = 0;
235 do {
236 ret = odp_queue_enq_multi(queue, &ev[enq_events],
237 events_per_queue -
238 enq_events);
239 if (ret < 0) {
240 ODPH_ERR("Queue enqueue failed.\n");
241 return -1;
242 }
243 enq_events += ret;
244 } while (enq_events < events_per_queue);
245
246 rdy_events += events_per_queue;
247 if (div_events && rdy_events >= tot_events)
248 return 0;
249 }
250 return 0;
251}

258static int output_results(test_globals_t *globals)
259{
260 test_stat_t *lat;
261 odp_schedule_sync_t stype;
262 test_stat_t total;
263 test_args_t *args;
264 uint64_t avg;
265 unsigned int i, j;
266
267 args = &globals->args;
268 stype = globals->args.sync_type;
269
270 printf("\n%s queue scheduling latency\n",
271 (stype == ODP_SCHED_SYNC_ATOMIC) ? "ATOMIC" :
272 ((stype == ODP_SCHED_SYNC_ORDERED) ? "ORDERED" : "PARALLEL"));
273
274 printf(" Forwarding mode: %s\n",
275 (args->forward_mode == EVENT_FORWARD_RAND) ? "random" :
276 ((args->forward_mode == EVENT_FORWARD_INC) ? "incremental" :
277 "none"));
278
279 printf(" LO_PRIO queues: %i\n", args->prio[LO_PRIO].queues);
280 if (args->prio[LO_PRIO].events_per_queue)
281 printf(" LO_PRIO event per queue: %i\n",
282 args->prio[LO_PRIO].events);
283 else
284 printf(" LO_PRIO events: %i\n", args->prio[LO_PRIO].events);
285
286 printf(" LO_PRIO sample events: %i\n", args->prio[LO_PRIO].sample_events);
287
288 printf(" HI_PRIO queues: %i\n", args->prio[HI_PRIO].queues);
289 if (args->prio[HI_PRIO].events_per_queue)
290 printf(" HI_PRIO event per queue: %i\n\n",
291 args->prio[HI_PRIO].events);
292 else
293 printf(" HI_PRIO events: %i\n", args->prio[HI_PRIO].events);
294
295 printf(" HI_PRIO sample events: %i\n\n", args->prio[HI_PRIO].sample_events);
296
297 if (globals->common_options.is_export) {
298 if (test_common_write("high priority avg (ns),high priority min (ns),"
299 "high priority max (ns),low priority avg (ns),"
300 "low priority min (ns),low priority max (ns)\n")) {
301 ODPH_ERR("Export failed\n");
302 test_common_write_term();
303 return -1;
304 }
305 }
306
307 for (i = 0; i < NUM_PRIOS; i++) {
308 memset(&total, 0, sizeof(test_stat_t));
309 total.min = UINT64_MAX;
310
311 printf("%s priority\n"
312 "Thread Avg[ns] Min[ns] Max[ns] Samples Total Max idx\n"
313 "-----------------------------------------------------------------------\n",
314 i == HI_PRIO ? "HIGH" : "LOW");
315 for (j = 1; j <= args->cpu_count; j++) {
316 lat = &globals->core_stat[j].prio[i];
317
318 if (lat->sample_events == 0) {
319 printf("%-8d N/A\n", j);
320 continue;
321 }
322
323 if (lat->max > total.max)
324 total.max = lat->max;
325 if (lat->min < total.min)
326 total.min = lat->min;
327 total.tot += lat->tot;
328 total.sample_events += lat->sample_events;
329 total.events += lat->events;
330
331 avg = lat->events ? lat->tot / lat->sample_events : 0;
332 printf("%-8d %-10" PRIu64 " %-10" PRIu64 " "
333 "%-10" PRIu64 " %-10" PRIu64 " %-10" PRIu64 " %-10" PRIu64 "\n",
334 j, avg, lat->min, lat->max, lat->sample_events,
335 lat->events, lat->max_idx);
336 }
337 printf("-----------------------------------------------------------------------\n");
338 if (total.sample_events == 0) {
339 printf("Total N/A\n\n");
340 continue;
341 }
342 avg = total.events ? total.tot / total.sample_events : 0;
343 printf("Total %-10" PRIu64 " %-10" PRIu64 " %-10" PRIu64 " "
344 "%-10" PRIu64 " %-10" PRIu64 "\n\n", avg, total.min,
345 total.max, total.sample_events, total.events);
346
347 if (globals->common_options.is_export) {
348 if (test_common_write("%" PRIu64 ",%" PRIu64 ",%" PRIu64 "%s",
349 avg, total.min, total.max, i == 0 ? "," : "")) {
350 ODPH_ERR("Export failed\n");
351 test_common_write_term();
352 return -1;
353 }
354 }
355 }
356
357 if (globals->common_options.is_export)
358 test_common_write_term();
359
360 return 0;
361}

363static int join_groups(test_globals_t *globals, int thr)
364{
365 odp_thrmask_t thrmask;
366 odp_schedule_group_t group;
367 int i, num;
368 int num_group = globals->args.num_group;
369
370 if (num_group <= 0)
371 return 0;
372
373 num = num_group;
374 if (globals->args.isolate)
375 num = 2 * num_group;
376
377 odp_thrmask_zero(&thrmask);
378 odp_thrmask_set(&thrmask, thr);
379
380 for (i = 0; i < num; i++) {
381 if (globals->args.isolate)
382 group = globals->group[i % 2][i / 2];
383 else
384 group = globals->group[0][i];
385
386 if (odp_schedule_group_join(group, &thrmask)) {
387 ODPH_ERR("Group join failed %i (thr %i)\n", i, thr);
388 return -1;
389 }
390 }
391
392 return 0;
393}

412static int test_schedule(int thr, test_globals_t *globals)
413{
414 odp_time_t time;
415 odp_event_t ev;
416 odp_buffer_t buf;
417 odp_queue_t dst_queue;
418 uint64_t latency;
419 uint64_t i;
420 test_event_t *event;
421 test_stat_t *stats;
422 int dst_idx, change_queue;
423 int warm_up_rounds = globals->args.warm_up_rounds;
424 uint64_t test_rounds = globals->args.test_rounds * (uint64_t)1000000;
425
426 memset(&globals->core_stat[thr], 0, sizeof(core_stat_t));
427 globals->core_stat[thr].prio[HI_PRIO].min = UINT64_MAX;
428 globals->core_stat[thr].prio[LO_PRIO].min = UINT64_MAX;
429
430 change_queue = globals->args.forward_mode != EVENT_FORWARD_NONE ? 1 : 0;
431
432 odp_barrier_wait(&globals->barrier);
433
434 for (i = 0; i < test_rounds; i++) {
435 ev = odp_schedule(NULL, ODP_SCHED_WAIT);
436
437 time = odp_time_global_strict();
438
439 buf = odp_buffer_from_event(ev);
440 event = odp_buffer_addr(buf);
441
442 stats = &globals->core_stat[thr].prio[event->prio];
443
444 if (event->type == SAMPLE) {
445 latency = odp_time_to_ns(time) - odp_time_to_ns(event->time_stamp);
446
447 if (latency > stats->max) {
448 stats->max = latency;
449 stats->max_idx = stats->sample_events;
450 }
451 if (latency < stats->min)
452 stats->min = latency;
453 stats->tot += latency;
454 stats->sample_events++;
455
456 /* Move sample event to a different priority */
457 if (!globals->args.sample_per_prio &&
458 globals->args.prio[!event->prio].queues)
459 event->prio = !event->prio;
460 }
461
462 if (odp_unlikely(event->type == WARM_UP)) {
463 event->warm_up_rounds++;
464 if (event->warm_up_rounds >= warm_up_rounds)
465 event->type = SAMPLE;
466 } else {
467 stats->events++;
468 }
469
470 /* Move event to next queue if forwarding is enabled */
471 if (change_queue)
472 dst_idx = event->src_idx[event->prio] + 1;
473 else
474 dst_idx = event->src_idx[event->prio];
475 if (dst_idx >= globals->args.prio[event->prio].queues)
476 dst_idx = 0;
477 event->src_idx[event->prio] = dst_idx;
478 dst_queue = globals->queue[event->prio][dst_idx];
479
480 if (event->type == SAMPLE)
481 event->time_stamp = odp_time_global_strict();
482
483 if (odp_queue_enq(dst_queue, ev)) {
484 ODPH_ERR("[%i] Queue enqueue failed.\n", thr);
485 odp_event_free(ev);
486 return -1;
487 }
488 }
489
490 /* Clear possible locally stored buffers */
491 odp_schedule_pause();
492
493 while (1) {
494 odp_queue_t src_queue;
495
496 ev = odp_schedule(&src_queue, ODP_SCHED_NO_WAIT);
497
498 if (ev == ODP_EVENT_INVALID)
499 break;
500
501 if (odp_queue_enq(src_queue, ev)) {
502 ODPH_ERR("[%i] Queue enqueue failed.\n", thr);
503 odp_event_free(ev);
504 return -1;
505 }
506 }
507
508 odp_barrier_wait(&globals->barrier);
509
510 if (thr == MAIN_THREAD) {
511 odp_schedule_resume();
512 clear_sched_queues(globals);
513 if (output_results(globals))
514 return -1;
515 }
516
517 return 0;
518}

528static int run_thread(void *arg ODP_UNUSED)
529{
530 odp_shm_t shm;
531 test_globals_t *globals;
532 test_args_t *args;
533 int thr;
534
535 thr = odp_thread_id();
536
537 shm = odp_shm_lookup("test_globals");
538 globals = odp_shm_addr(shm);
539
540 if (globals == NULL) {
541 ODPH_ERR("Shared mem lookup failed\n");
542 return -1;
543 }
544
545 if (join_groups(globals, thr))
546 return -1;
547
548 if (thr == MAIN_THREAD) {
549 args = &globals->args;
550
551 if (enqueue_events(HI_PRIO, args->prio[HI_PRIO].queues,
552 args->prio[HI_PRIO].events, args->prio[HI_PRIO].sample_events,
553 !args->prio[HI_PRIO].events_per_queue,
554 globals))
555 return -1;
556
557 if (enqueue_events(LO_PRIO, args->prio[LO_PRIO].queues,
558 args->prio[LO_PRIO].events, args->prio[LO_PRIO].sample_events,
559 !args->prio[LO_PRIO].events_per_queue,
560 globals))
561 return -1;
562 }
563
564 if (test_schedule(thr, globals))
565 return -1;
566
567 return 0;
568}

620static void parse_args(int argc, char *argv[], test_args_t *args)
621{
622 int opt;
623 int i;
624
625 static const struct option longopts[] = {
626 {"count", required_argument, NULL, 'c'},
627 {"duration", required_argument, NULL, 'd'},
628 {"forward-mode", required_argument, NULL, 'f'},
629 {"num_group", required_argument, NULL, 'g'},
630 {"isolate", required_argument, NULL, 'i'},
631 {"lo-prio-queues", required_argument, NULL, 'l'},
632 {"hi-prio-queues", required_argument, NULL, 't'},
633 {"lo-prio-events-per-queue", required_argument, NULL, 'm'},
634 {"hi-prio-events-per-queue", required_argument, NULL, 'n'},
635 {"lo-prio-events", required_argument, NULL, 'o'},
636 {"hi-prio-events", required_argument, NULL, 'p'},
637 {"sync", required_argument, NULL, 's'},
638 {"warm-up", required_argument, NULL, 'w'},
639 {"sample-per-prio", no_argument, NULL, 'r'},
640 {"help", no_argument, NULL, 'h'},
641 {NULL, 0, NULL, 0}
642 };
643
644 static const char *shortopts = "+c:d:f:g:i:l:t:m:n:o:p:s:w:rh";
645
646 args->cpu_count = 1;
647 args->forward_mode = EVENT_FORWARD_RAND;
648 args->num_group = 0;
649 args->isolate = 0;
650 args->test_rounds = 10;
651 args->warm_up_rounds = 100;
652 args->sync_type = ODP_SCHED_SYNC_PARALLEL;
653 args->sample_per_prio = 0;
654 args->prio[LO_PRIO].queues = 64;
655 args->prio[HI_PRIO].queues = 16;
656 args->prio[LO_PRIO].events = 32;
657 args->prio[HI_PRIO].events = 0;
658 args->prio[LO_PRIO].events_per_queue = 1;
659 args->prio[HI_PRIO].events_per_queue = 0;
660 args->prio[LO_PRIO].sample_events = 0;
661 args->prio[HI_PRIO].sample_events = 1;
662
663 while (1) {
664 opt = getopt_long(argc, argv, shortopts, longopts, NULL);
665
666 if (opt == -1)
667 break; /* No more options */
668
669 switch (opt) {
670 case 'c':
671 args->cpu_count = atoi(optarg);
672 break;
673 case 'd':
674 args->test_rounds = atoi(optarg);
675 break;
676 case 'f':
677 args->forward_mode = atoi(optarg);
678 break;
679 case 'g':
680 args->num_group = atoi(optarg);
681 break;
682 case 'i':
683 args->isolate = atoi(optarg);
684 break;
685 case 'l':
686 args->prio[LO_PRIO].queues = atoi(optarg);
687 break;
688 case 't':
689 args->prio[HI_PRIO].queues = atoi(optarg);
690 break;
691 case 'm':
692 args->prio[LO_PRIO].events = atoi(optarg);
693 args->prio[LO_PRIO].events_per_queue = 1;
694 break;
695 case 'n':
696 args->prio[HI_PRIO].events = atoi(optarg);
697 args->prio[HI_PRIO].events_per_queue = 1;
698 break;
699 case 'o':
700 args->prio[LO_PRIO].events = atoi(optarg);
701 args->prio[LO_PRIO].events_per_queue = 0;
702 break;
703 case 'p':
704 args->prio[HI_PRIO].events = atoi(optarg);
705 args->prio[HI_PRIO].events_per_queue = 0;
706 break;
707 case 's':
708 i = atoi(optarg);
709 if (i == 1)
710 args->sync_type = ODP_SCHED_SYNC_ATOMIC;
711 else if (i == 2)
712 args->sync_type = ODP_SCHED_SYNC_ORDERED;
713 else
714 args->sync_type = ODP_SCHED_SYNC_PARALLEL;
715 break;
716 case 'r':
717 args->sample_per_prio = 1;
718 break;
719 case 'w':
720 args->warm_up_rounds = atoi(optarg);
721 break;
722 case 'h':
723 usage();
724 exit(EXIT_SUCCESS);
725 break;
726
727 default:
728 break;
729 }
730 }
731
732 /* Make sure arguments are valid */
733 /* -1 for main thread */
734 if (args->cpu_count > ODP_THREAD_COUNT_MAX - 1)
735 args->cpu_count = ODP_THREAD_COUNT_MAX - 1;
736 if (args->prio[LO_PRIO].queues > MAX_QUEUES)
737 args->prio[LO_PRIO].queues = MAX_QUEUES;
738 if (args->prio[HI_PRIO].queues > MAX_QUEUES)
739 args->prio[HI_PRIO].queues = MAX_QUEUES;
740 if (args->test_rounds < 1)
741 args->test_rounds = 1;
742 if (!args->prio[HI_PRIO].queues && !args->prio[LO_PRIO].queues) {
743 printf("No queues configured\n");
744 usage();
745 exit(EXIT_FAILURE);
746 }
747 if (args->forward_mode > EVENT_FORWARD_NONE ||
748 args->forward_mode < EVENT_FORWARD_RAND) {
749 printf("Invalid forwarding mode\n");
750 usage();
751 exit(EXIT_FAILURE);
752 }

763static void randomize_queues(odp_queue_t queues[], uint32_t num, uint64_t *seed)
764{
765 uint32_t i;
766
767 for (i = 0; i < num; i++) {
768 uint32_t new_index;
769 odp_queue_t swap_queue;
770 odp_queue_t cur_queue = queues[i];
771
772 odp_random_test_data((uint8_t *)&new_index, sizeof(new_index),
773 seed);
774 new_index = new_index % num;
775 swap_queue = queues[new_index];
776
777 queues[new_index] = cur_queue;
778 queues[i] = swap_queue;
779 }
780}

844static int calc_queue_sizes(test_globals_t *globals, uint32_t queue_size[])
845{
846 odp_schedule_capability_t capa;
847 test_args_t *args = &globals->args;
848 const uint32_t min_queue_size = 256;
849 uint32_t tot_queues = 0;
850
851 if (odp_schedule_capability(&capa)) {
852 ODPH_ERR("Schedule capability failed\n");
853 return -1;
854 }
855
856 for (int i = 0; i < NUM_PRIOS; i++) {
857 uint32_t events = args->prio[i].events;
858 int queues = args->prio[i].queues;
859
860 if (!args->prio[i].events_per_queue && queues)
861 events = (events + queues - 1) / queues;
862
863 /* Events may stack up if forwarding is enabled */
864 if (args->forward_mode != EVENT_FORWARD_NONE)
865 events *= queues;
866
867 /* Reserve room for sample event */
868 events++;
869
870 queue_size[i] = ODPH_MAX(events, min_queue_size);
871
872 if (capa.max_queue_size && queue_size[i] > capa.max_queue_size) {
873 ODPH_ERR("Warn: queues may not be able to store all events (required size "
874 "%" PRIu32 ", max supported %" PRIu32 ")\n", queue_size[i],
875 capa.max_queue_size);
876 queue_size[i] = capa.max_queue_size;
877 }
878 tot_queues += queues;
879 }
880
881 if (tot_queues > capa.max_queues) {
882 ODPH_ERR("Requested %" PRIu32 " queues, max %" PRIu32 " supported\n",
883 tot_queues, capa.max_queues);
884 return -1;
885 }
886
887 return 0;
888}

890static int create_queues(test_globals_t *globals)
891{
892 odp_queue_param_t param;
893 test_args_t *args = &globals->args;
894 int num_group = args->num_group;
895 uint32_t queue_size[NUM_PRIOS];
896
897 if (calc_queue_sizes(globals, queue_size))
898 return -1;
899
900 odp_queue_param_init(&param);
901 param.type = ODP_QUEUE_TYPE_SCHED;
902 param.sched.sync = args->sync_type;
903
904 for (int i = 0; i < NUM_PRIOS; i++) {
905 char name[] = "sched_XX_YY";
906 odp_queue_t queue;
907 odp_schedule_group_t grp = num_group < 0 ? ODP_SCHED_GROUP_WORKER :
908 ODP_SCHED_GROUP_ALL;
909 const int prio = i == HI_PRIO ? odp_schedule_max_prio() :
910 odp_schedule_min_prio();
911
912 param.sched.prio = prio;
913 param.size = queue_size[i];
914
915 /* Replace XX and YY in name to differentiate queues */
916 name[6] = '0' + (prio / 10);
917 name[7] = '0' + prio - (10 * (prio / 10));
918
919 for (int j = 0; j < args->prio[i].queues; j++) {
920 name[9] = '0' + j / 10;
921 name[10] = '0' + j - 10 * (j / 10);
922
923 /* Round robin queues into groups */
924 if (num_group > 0)
925 grp = globals->group[i][j % num_group];
926
927 param.sched.group = grp;
928
929 queue = odp_queue_create(name, &param);
930
931 if (queue == ODP_QUEUE_INVALID) {
932 ODPH_ERR("Scheduled queue create failed\n");
933 return -1;
934 }
935
936 globals->queue[i][j] = queue;
937 }
938 if (args->forward_mode == EVENT_FORWARD_RAND) {
939 uint64_t seed = i;
940
941 randomize_queues(globals->queue[i], args->prio[i].queues, &seed);
942 }
943 }
944 return 0;
945}

1006 ODPH_ERR("ODP global init failed.\n");
1007 exit(EXIT_FAILURE);

1010 odp_sys_info_print();

1033 printf(" Isolated groups: %i\n", args.isolate);

1118
1119 for (i = 0; i < NUM_PRIOS; i++) {
1120 odp_queue_t queue;
1121 int num_queues;
1122
1123 num_queues = args.prio[i].queues;
1124
1125 for (j = 0; j < num_queues; j++) {
1126 queue = globals->queue[i][j];
1127 if (odp_queue_destroy(queue)) {
1128 ODPH_ERR("Queue destroy failed [%i][%i]\n", i, j);
1129 err = -1;
1130 break;
1131 }
1132 }
1133 }
1134
1135error:
1136 if (destroy_groups(group, tot_group)) {
1137 ODPH_ERR("Group destroy failed\n");
1138 err = -1;
1139 }
1140
1141 if (pool != ODP_POOL_INVALID) {
1142 if (odp_pool_destroy(pool)) {
1143 ODPH_ERR("Pool destroy failed\n");
1144 err = -1;
1145 }
1146 }
1147
1148 if (shm != ODP_SHM_INVALID) {
1149 if (odp_shm_free(shm)) {
1150 ODPH_ERR("SHM destroy failed\n");
1151 err = -1;
1152 }
1153 }
1154
1155 err += odp_term_local();
1156 err += odp_term_global(instance);
1157
1158 return err;
1159}
