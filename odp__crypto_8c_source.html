543static unsigned long long
544get_rusage_diff(struct rusage *start, struct rusage *end)
545{
546 unsigned long long rusage_diff;
547 unsigned long long rusage_start;
548 unsigned long long rusage_end;
549
550 rusage_start = (start->ru_utime.tv_sec * 1000000) +
551 (start->ru_utime.tv_usec);
552 rusage_start += (start->ru_stime.tv_sec * 1000000) +
553 (start->ru_stime.tv_usec);
554
555 rusage_end = (end->ru_utime.tv_sec * 1000000) +
556 (end->ru_utime.tv_usec);
557 rusage_end += (end->ru_stime.tv_sec * 1000000) +
558 (end->ru_stime.tv_usec);
559
560 rusage_diff = rusage_end - rusage_start;
561
562 return rusage_diff;
563}

637static void
638print_mem(const char *msg,
639 const unsigned char *ptr,
640 unsigned int len)
641{
642 unsigned i, j;
643 char c;
644 char line[81];
645 char *p;
646
647 if (msg)
648 printf("\n%s (bytes size = %d)", msg, len);
649
650 for (i = 0; i < len; i += 16) {
651 p = line;
652 sprintf(p, "\n%04x ", i); p += 8;
653
654 for (j = 0; j < 16; j++) {
655 if (i + j == len)
656 break;
657
658 sprintf(p, " %02x", (ptr)[i + j]); p += 3;
659 }
660
661 for (; j < 16; j++) {
662 sprintf(p, " "); p += 3;
663 }
664
665 sprintf(p, " "); p += 3;
666
667 for (j = 0; j < 16; j++) {
668 if (i + j == len)
669 break;
670 c = (ptr)[i + j];
671 *p++ = (' ' <= c && c <= '~') ? c : '.';
672 }
673
674 *p = '\0';
675 printf("%s", line);
676 }
677 printf("\n");
678}

683static int
684create_session_from_config(odp_crypto_session_t *session,
685 crypto_alg_config_t *config,
686 crypto_args_t *cargs)
687{
688 odp_crypto_session_param_t params;
689 odp_crypto_ses_create_err_t ses_create_rc;
690 odp_queue_t out_queue;
691
692 memcpy(&params, &config->session, sizeof(odp_crypto_session_param_t));
693 params.op = ODP_CRYPTO_OP_ENCODE;
694 params.op_type = ODP_CRYPTO_OP_TYPE_BASIC;
695 params.auth_cipher_text = true;
696 params.output_pool = ODP_POOL_INVALID;
697
698 if (cargs->schedule || cargs->poll) {
699 out_queue = odp_queue_lookup("crypto-out");
700 if (out_queue == ODP_QUEUE_INVALID) {
701 ODPH_ERR("crypto-out queue not found\n");
702 return -1;
703 }
704 params.compl_queue = out_queue;
705 params.op_mode = ODP_CRYPTO_ASYNC;
706 } else {
707 params.compl_queue = ODP_QUEUE_INVALID;
708 params.op_mode = ODP_CRYPTO_SYNC;
709 }
710 if (odp_crypto_session_create(&params, session,
711 &ses_create_rc)) {
712 switch (ses_create_rc) {
713 case ODP_CRYPTO_SES_ERR_ALG_COMBO:
714 printf(" requested algorithm combination not supported\n");
715 return 1;
716 case ODP_CRYPTO_SES_ERR_ALG_ORDER:
717 printf(" requested algorithm order not supported\n");
718 return 1;
719 case ODP_CRYPTO_SES_ERR_PARAMS:
720 printf(" requested session parameters not supported\n");
721 return 1;
722 default:
723 break;
724 }
725 ODPH_ERR("crypto session create failed.\n");
726 return -1;
727 }
728
729 return 0;
730}

754static int
755run_measure_one(crypto_args_t *cargs,
756 crypto_alg_config_t *config,
757 odp_crypto_session_t *session,
758 unsigned int payload_length,
759 crypto_run_result_t *result)
760{
761 odp_crypto_packet_op_param_t params;
762
763 odp_pool_t pkt_pool;
764 odp_queue_t out_queue;
765 odp_packet_t pkt = ODP_PACKET_INVALID;
766 int rc = 0;
767 uint32_t packet_len = payload_length + MAX_AUTH_DIGEST_LEN;
768
769 pkt_pool = odp_pool_lookup("packet_pool");
770 if (pkt_pool == ODP_POOL_INVALID) {
771 ODPH_ERR("pkt_pool not found\n");
772 return -1;
773 }
774
775 out_queue = odp_queue_lookup("crypto-out");
776 if (cargs->schedule || cargs->poll) {
777 if (out_queue == ODP_QUEUE_INVALID) {
778 ODPH_ERR("crypto-out queue not found\n");
779 return -1;
780 }
781 }
782
783 if (cargs->reuse_packet) {
784 pkt = make_packet(pkt_pool, packet_len);
785 if (ODP_PACKET_INVALID == pkt)
786 return -1;
787 }
788
789 time_record_t start, end;
790 int packets_sent = 0;
791 int packets_received = 0;
792
793 /* Initialize parameters block */
794 memset(&params, 0, sizeof(params));
795 params.session = *session;
796 params.cipher_iv_ptr = test_iv;
797 params.auth_iv_ptr = test_iv;
798 params.aad_ptr = test_aad;
799
800 params.cipher_range.offset = 0;
801 params.cipher_range.length = config->cipher_in_bit_mode ? payload_length * 8
802 : payload_length;
803 params.auth_range.offset = 0;
804 params.auth_range.length = config->auth_in_bit_mode ? payload_length * 8
805 : payload_length;
806 params.hash_result_offset = payload_length;
807
808 fill_time_record(&start);
809
810 while ((packets_sent < cargs->iteration_count) ||
811 (packets_received < cargs->iteration_count)) {
812 void *mem;
813
814 if ((packets_sent < cargs->iteration_count) &&
815 (packets_sent - packets_received <
816 cargs->in_flight)) {
817 odp_packet_t out_pkt;
818
819 if (!cargs->reuse_packet) {
820 pkt = make_packet(pkt_pool, packet_len);
821 if (ODP_PACKET_INVALID == pkt)
822 return -1;
823 }
824
825 if (cargs->debug_packets) {
826 mem = odp_packet_data(pkt);
827 print_mem("Packet before encryption:",
828 mem, payload_length);
829 }
830
831 if (cargs->schedule || cargs->poll) {
832 rc = odp_crypto_op_enq(&pkt, &out_pkt,
833 &params, 1);
834 if (rc <= 0) {
835 ODPH_ERR("failed odp_crypto_packet_op_enq: rc = %d\n", rc);
836 if (!cargs->reuse_packet)
837 odp_packet_free(pkt);
838 break;
839 }
840 packets_sent += rc;
841 } else {
842 rc = odp_crypto_op(&pkt, &out_pkt,
843 &params, 1);
844 if (rc <= 0) {
845 ODPH_ERR("failed odp_crypto_packet_op: rc = %d\n", rc);
846 if (!cargs->reuse_packet)
