129 static int enqueue_events(int thr, int prio, int num_queues, int num_events,
130  test_globals_t *globals)
131 {
132  odp_buffer_t buf[num_events];
133  odp_event_t ev[num_events];
134  odp_queue_t queue;
135  int i, j, k, ret;
136 
137  if (prio == odp_schedule_max_prio())
138  i = 0;
139  else
140  i = 1;
141 
142  /* Alloc and enqueue a buffer per queue */
143  for (j = 0; j < num_queues; j++) {
144  queue = globals->queue[i][j];
145 
146  ret = odp_buffer_alloc_multi(globals->pool, buf, num_events);
147  if (ret != num_events) {
148  ODPH_ERR(" [%i] buffer alloc failed\n", thr);
149  ret = ret < 0 ? 0 : ret;
150  ret = ret > num_events ? num_events : ret; /* GCC-9 -O3 workaround */
151  odp_buffer_free_multi(buf, ret);
152  return -1;
153  }
154  for (k = 0; k < num_events; k++) {
155  if (!odp_buffer_is_valid(buf[k])) {
156  ODPH_ERR(" [%i] buffer alloc failed\n", thr);
157  odp_buffer_free_multi(buf, num_events);
158  return -1;
159  }
160  ev[k] = odp_buffer_to_event(buf[k]);
161  }
162 
163  ret = odp_queue_enq_multi(queue, ev, num_events);
164  if (ret != num_events) {
165  ODPH_ERR(" [%i] Queue enqueue failed.\n", thr);
166  ret = ret < 0 ? 0 : ret;
167  odp_buffer_free_multi(&buf[ret], num_events - ret);
168  return -1;
169  }
170  }
171 
172  return 0;
173 }

220 static int test_alloc_multi(int thr, test_globals_t *globals)
221 {
222  int i, j, ret;
223  const int num_alloc = MAX_ALLOCS;
224  odp_buffer_t temp_buf[num_alloc];
225  uint64_t c1, c2, cycles;
226 
227  c1 = odp_cpu_cycles();
228 
229  for (i = 0; i < ALLOC_ROUNDS; i++) {
230  ret = odp_buffer_alloc_multi(globals->pool, temp_buf, num_alloc);
231  if (ret != num_alloc) {
232  ODPH_ERR(" [%i] buffer alloc failed\n", thr);
233  ret = ret < 0 ? 0 : ret;
234  odp_buffer_free_multi(temp_buf, ret);
235  return -1;
236  }
237 
238  for (j = 0; j < num_alloc; j++) {
239  if (!odp_buffer_is_valid(temp_buf[j])) {
240  ODPH_ERR(" [%i] alloc_multi failed\n", thr);
241  odp_buffer_free_multi(temp_buf, num_alloc);
242  return -1;
243  }
244  }
245  odp_buffer_free_multi(temp_buf, num_alloc);
246  }
247 
248  c2 = odp_cpu_cycles();
249  cycles = odp_cpu_cycles_diff(c2, c1);
250  cycles = cycles / (ALLOC_ROUNDS * num_alloc);
251 
252  printf(" [%i] alloc_multi alloc+free %6" PRIu64 " CPU cycles\n",
253  thr, cycles);
254 
255  return 0;

268 static int test_plain_queue(int thr, test_globals_t *globals)
269 {
270  odp_event_t ev;
271  odp_buffer_t buf;
272  test_message_t *t_msg;
273  odp_queue_t queue;
274  uint64_t c1, c2, cycles;
275  int i, j;
276 
277  /* Alloc test message */
278  buf = odp_buffer_alloc(globals->pool);
279 
280  if (!odp_buffer_is_valid(buf)) {
281  ODPH_ERR(" [%i] buffer alloc failed\n", thr);
282  return -1;
283  }
284 
285  /* odp_buffer_print(buf); */
286 
287  t_msg = odp_buffer_addr(buf);
288  t_msg->msg_id = MSG_HELLO;
289  t_msg->seq = 0;
290 
291  queue = odp_queue_lookup("plain_queue");
292 
293  if (queue == ODP_QUEUE_INVALID) {
294  printf(" [%i] Queue lookup failed.\n", thr);
295  return -1;
296  }
297 
298  c1 = odp_cpu_cycles();
299 
300  for (i = 0; i < QUEUE_ROUNDS; i++) {
301  ev = odp_buffer_to_event(buf);
302 
303  if (odp_queue_enq(queue, ev)) {
304  ODPH_ERR(" [%i] Queue enqueue failed.\n", thr);
305  odp_buffer_free(buf);
306  return -1;
307  }
308 
309  /* When enqueue and dequeue are decoupled (e.g. not using a
310  * common lock), an enqueued event may not be immediately
311  * visible to dequeue. So we just try again for a while. */
312  for (j = 0; j < 100; j++) {
313  ev = odp_queue_deq(queue);
314  if (ev != ODP_EVENT_INVALID)
315  break;
316  odp_cpu_pause();
317  }
318 
319  buf = odp_buffer_from_event(ev);
320 
321  if (!odp_buffer_is_valid(buf)) {
322  ODPH_ERR(" [%i] Queue empty.\n", thr);
323  return -1;
324  }
325  }
326 
327  c2 = odp_cpu_cycles();
328  cycles = odp_cpu_cycles_diff(c2, c1);
329  cycles = cycles / QUEUE_ROUNDS;
330 
331  printf(" [%i] plain_queue enq+deq %6" PRIu64 " CPU cycles\n",
332  thr, cycles);
333 
334  odp_buffer_free(buf);
335  return 0;
336 }

351 static int test_schedule_single(const char *str, int thr,
352  int prio, test_globals_t *globals)
353 {
354  odp_event_t ev;
355  odp_queue_t queue;
356  uint64_t c1, c2, cycles;
357  uint32_t i;
358  uint32_t tot;
359 
360  if (enqueue_events(thr, prio, 1, 1, globals))
361  return -1;
362 
363  c1 = odp_cpu_cycles();
364 
365  for (i = 0; i < QUEUE_ROUNDS; i++) {
366  ev = odp_schedule(&queue, ODP_SCHED_WAIT);
367 
368  if (odp_queue_enq(queue, ev)) {
369  ODPH_ERR(" [%i] Queue enqueue failed.\n", thr);
370  odp_event_free(ev);
371  return -1;
372  }
373  }
374 
375  /* Clear possible locally stored buffers */
376  odp_schedule_pause();
377 
378  tot = i;
379 
380  while (1) {
381  ev = odp_schedule(&queue, ODP_SCHED_NO_WAIT);
382 
383  if (ev == ODP_EVENT_INVALID)
384  break;
385 
386  tot++;
387 
388  if (odp_queue_enq(queue, ev)) {
389  ODPH_ERR(" [%i] Queue enqueue failed.\n", thr);
390  odp_event_free(ev);
391  return -1;
392  }
393  }
394 
395  c2 = odp_cpu_cycles();
396  cycles = odp_cpu_cycles_diff(c2, c1);
397 
398  odp_barrier_wait(&globals->barrier);
399 
400  odp_schedule_resume();
401 
402  clear_sched_queues();
403 
404  cycles = cycles / tot;
405 
406  printf(" [%i] %s enq+deq %6" PRIu64 " CPU cycles\n", thr, str, cycles);
407 
408  return 0;
409 }

495 static int test_schedule_multi(const char *str, int thr,
496  int prio, test_globals_t *globals)
497 {
498  odp_event_t ev[MULTI_BUFS_MAX];
499  odp_queue_t queue;
500  uint64_t c1, c2, cycles;
501  int i;
502  int num;
503  uint32_t tot = 0;
504 
505  if (enqueue_events(thr, prio, globals->queues_per_prio, MULTI_BUFS_MAX,
506  globals))
507  return -1;
508 
509  /* Start sched-enq loop */
510  c1 = odp_cpu_cycles();
511 
512  for (i = 0; i < QUEUE_ROUNDS; i++) {
513  num = odp_schedule_multi(&queue, ODP_SCHED_WAIT, ev,
514  MULTI_BUFS_MAX);
515 
516  tot += num;
517 
518  if (globals->args.fairness) {
519  queue_context_t *queue_ctx;
520 
521  queue_ctx = odp_queue_context(queue);
522  queue_ctx->num_ev += num;
523  }
524 
525  /* Assume we can enqueue all events */
526  if (odp_queue_enq_multi(queue, ev, num) != num) {
527  ODPH_ERR(" [%i] Queue enqueue failed.\n", thr);
528  return -1;
529  }
530  }
531 
532  /* Clear possible locally stored events */
533  odp_schedule_pause();
534 
535  while (1) {
536  num = odp_schedule_multi(&queue, ODP_SCHED_NO_WAIT, ev,
537  MULTI_BUFS_MAX);
538 
539  if (num == 0)
540  break;
541 
542  tot += num;
543 
544  if (globals->args.fairness) {
545  queue_context_t *queue_ctx;
546 
547  queue_ctx = odp_queue_context(queue);
548  queue_ctx->num_ev += num;
549  }
550 
551  /* Assume we can enqueue all events */
552  if (odp_queue_enq_multi(queue, ev, num) != num) {
553  ODPH_ERR(" [%i] Queue enqueue failed.\n", thr);
554  return -1;
555  }
556  }
557 
558  c2 = odp_cpu_cycles();
559  cycles = odp_cpu_cycles_diff(c2, c1);
560 
561  odp_barrier_wait(&globals->barrier);
562 
563  odp_schedule_resume();
564 
565  clear_sched_queues();
566 
567  if (tot)
568  cycles = cycles / tot;
569  else
570  cycles = 0;
571 
572  printf(" [%i] %s enq+deq %6" PRIu64 " CPU cycles\n", thr, str, cycles);
573 
574  odp_barrier_wait(&globals->barrier);
575 
576  if (globals->args.fairness && globals->first_thr == thr)
577  print_stats(prio, globals);
578 
579  return 0;
580 }

589 static int run_thread(void *arg ODP_UNUSED)
590 {
591  int thr;
592  odp_shm_t shm;
593  test_globals_t *globals;
594  odp_barrier_t *barrier;
595 
596  thr = odp_thread_id();
597 
598  printf("Thread %i starts on CPU %i\n", thr, odp_cpu_id());
599 
600  shm = odp_shm_lookup("test_globals");
601  globals = odp_shm_addr(shm);
602 
603  if (globals == NULL) {
604  ODPH_ERR("Shared mem lookup failed\n");
605  return -1;
606  }
607 
608  barrier = &globals->barrier;
609 
610  /*
611  * Test barriers back-to-back
612  */
613  odp_barrier_wait(barrier);
614  odp_barrier_wait(barrier);
615  odp_barrier_wait(barrier);
616  odp_barrier_wait(barrier);
617  odp_barrier_wait(barrier);
618 
619  /* Select which thread is the first_thr */
620  while (globals->first_thr < 0) {
621  if (odp_spinlock_trylock(&globals->lock)) {
622  globals->first_thr = thr;
623  odp_spinlock_unlock(&globals->lock);
624  }
625  }
626 
627  odp_barrier_wait(barrier);
628 
629  if (test_alloc_single(thr, globals))
630  return -1;
631 
632  odp_barrier_wait(barrier);
633 
634  if (test_alloc_multi(thr, globals))
635  return -1;
636 
637  odp_barrier_wait(barrier);
638 
639  if (test_plain_queue(thr, globals))
640  return -1;
641 
642  /* Low prio */
643 
644  odp_barrier_wait(barrier);
645 
646  if (test_schedule_single("sched_____s_lo", thr,
647  odp_schedule_min_prio(), globals))
648  return -1;
649 
650  odp_barrier_wait(barrier);
651 
652  if (test_schedule_many("sched_____m_lo", thr,
653  odp_schedule_min_prio(), globals))
654  return -1;
655 
656  odp_barrier_wait(barrier);
657 
658  if (test_schedule_multi("sched_multi_lo", thr,
659  odp_schedule_min_prio(), globals))
660  return -1;
661 
662  /* High prio */
663 
664  odp_barrier_wait(barrier);
665 
666  if (test_schedule_single("sched_____s_hi", thr,
667  odp_schedule_max_prio(), globals))
668  return -1;
669 
670  odp_barrier_wait(barrier);
671 
672  if (test_schedule_many("sched_____m_hi", thr,
673  odp_schedule_max_prio(), globals))
674  return -1;
675 
676  odp_barrier_wait(barrier);
677 
678  if (test_schedule_multi("sched_multi_hi", thr,
679  odp_schedule_max_prio(), globals))
680  return -1;
681 
682  printf("Thread %i exits\n", thr);
683  fflush(NULL);
684  return 0;
685 }

690 static void test_cpu_freq(double test_sec)
691 {
692  odp_time_t cur_time, test_time, start_time, end_time;
693  uint64_t c1, c2, cycles;
694  uint64_t nsec;
695  double diff_max_hz, max_cycles;
696 
697  printf("\nCPU cycle count frequency test (runs about %f sec)\n",
698  test_sec);
699 
700  test_time = odp_time_local_from_ns(test_sec * ODP_TIME_SEC_IN_NS);
701  start_time = odp_time_local();
702  end_time = odp_time_sum(start_time, test_time);
703 
704  /* Start the measurement */
705  c1 = odp_cpu_cycles();
706 
707  do {
708  cur_time = odp_time_local();
709  } while (odp_time_cmp(end_time, cur_time) > 0);
710 
711  c2 = odp_cpu_cycles();
712 
713  test_time = odp_time_diff(cur_time, start_time);
714  nsec = odp_time_to_ns(test_time);
715 
716  cycles = odp_cpu_cycles_diff(c2, c1);
717  max_cycles = (nsec * odp_cpu_hz_max()) / 1000000000.0;
718 
719  /* Compare measured CPU cycles to maximum theoretical CPU cycle count */
720  diff_max_hz = ((double)(cycles) - max_cycles) / max_cycles;
721 
722  printf("odp_time %" PRIu64 " ns\n", nsec);
723  printf("odp_cpu_cycles %" PRIu64 " CPU cycles\n", cycles);
724  printf("odp_sys_cpu_hz %" PRIu64 " hz\n", odp_cpu_hz_max());
725  printf("Diff from max CPU freq %f%%\n", diff_max_hz * 100.0);
726 
727  printf("\n");
728 }
