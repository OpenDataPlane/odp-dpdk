192 static int scan_ip(char *buf, unsigned int *paddr)
193 {
194  unsigned int part1, part2, part3, part4;
195  char tail = 0;
196  int field;
197 
198  if (buf == NULL)
199  return 0;
200 
201  field = sscanf(buf, "%u . %u . %u . %u %c",
202  &part1, &part2, &part3, &part4, &tail);
203 
204  if (field < 4 || field > 5) {
205  printf("expect 4 field,get %d/n", field);
206  return 0;
207  }
208 
209  if (tail != 0) {
210  printf("ip address mixed with non number/n");
211  return 0;
212  }
213 
214  if (part1 <= 255 && part2 <= 255 && part3 <= 255 && part4 <= 255) {
215  if (paddr)
216  *paddr = part1 << 24 | part2 << 16 | part3 << 8 | part4;
217  return 1;
218  }
219 
220  printf("not good ip %u:%u:%u:%u/n", part1, part2, part3, part4);
221 
222  return 0;
223 }

305 static odp_packet_t setup_udp_pkt_ref(odp_pool_t pool,
306  odp_pktout_config_opt_t *pktout_cfg)
307 {
308  odp_packet_t pkt;
309  char *buf;
310  odph_ethhdr_t *eth;
311  odph_ipv4hdr_t *ip;
312  odph_udphdr_t *udp;
313 
314  pkt = odp_packet_alloc(pool, args->appl.payload + ODPH_UDPHDR_LEN +
315  ODPH_IPV4HDR_LEN + ODPH_ETHHDR_LEN);
316 
317  if (pkt == ODP_PACKET_INVALID)
318  return pkt;
319 
320  buf = odp_packet_data(pkt);
321 
322  /* ether */
323  odp_packet_l2_offset_set(pkt, 0);
324  eth = (odph_ethhdr_t *)buf;
325  memcpy((char *)eth->src.addr, args->appl.srcmac.addr, ODPH_ETHADDR_LEN);
326  memcpy((char *)eth->dst.addr, args->appl.dstmac.addr, ODPH_ETHADDR_LEN);
327  eth->type = odp_cpu_to_be_16(ODPH_ETHTYPE_IPV4);
328 
329  /* ip */
330  odp_packet_l3_offset_set(pkt, ODPH_ETHHDR_LEN);
331  odp_packet_has_ipv4_set(pkt, 1);
332  ip = (odph_ipv4hdr_t *)(buf + ODPH_ETHHDR_LEN);
333  ip->dst_addr = odp_cpu_to_be_32(args->appl.dstip);
334  ip->src_addr = odp_cpu_to_be_32(args->appl.srcip);
335  ip->ver_ihl = ODPH_IPV4 << 4 | ODPH_IPV4HDR_IHL_MIN;
336  ip->tot_len = odp_cpu_to_be_16(args->appl.payload + ODPH_UDPHDR_LEN +
337  ODPH_IPV4HDR_LEN);
338  ip->proto = ODPH_IPPROTO_UDP;
339  ip->id = 0;
340  ip->ttl = 64;
341  ip->chksum = 0;
342 
343  /* udp */
344  odp_packet_l4_offset_set(pkt, ODPH_ETHHDR_LEN + ODPH_IPV4HDR_LEN);
345  odp_packet_has_udp_set(pkt, 1);
346  udp = (odph_udphdr_t *)(buf + ODPH_ETHHDR_LEN + ODPH_IPV4HDR_LEN);
347  udp->src_port = odp_cpu_to_be_16(args->appl.srcport);
348  udp->dst_port = odp_cpu_to_be_16(args->appl.dstport);
349  udp->length = odp_cpu_to_be_16(args->appl.payload + ODPH_UDPHDR_LEN);
350  if (!pktout_cfg->bit.udp_chksum) {
351  udp->chksum = 0;
352  udp->chksum = odph_ipv4_udp_chksum(pkt);
353  }
354 
355  return pkt;
356 }

367 static int setup_udp_pkt(odp_packet_t pkt, odp_pktout_config_opt_t *pktout_cfg,
368  counters_t *counters, void *arg)
369 {
370  char *buf;
371  odph_ipv4hdr_t *ip;
372  unsigned short seq;
373  udp_args_t *udp_arg = (udp_args_t *)arg;
374 
375  buf = (char *)odp_packet_data(pkt);
376 
377  /*Update IP ID and checksum*/
378  ip = (odph_ipv4hdr_t *)(buf + ODPH_ETHHDR_LEN);
379  seq = counters->ctr_seq % 0xFFFF;
380  counters->ctr_seq++;
381  ip->id = odp_cpu_to_be_16(seq);
382  if (!pktout_cfg->bit.ipv4_chksum) {
383  ip->chksum = 0;
384  ip->chksum = ~odp_chksum_ones_comp16(ip, ODPH_IPV4HDR_LEN);
385  }
386 
387  if (udp_arg->multi_flow) {
388  odph_udphdr_t *udp = (odph_udphdr_t *)(buf + ODPH_ETHHDR_LEN +
389  ODPH_IPV4HDR_LEN);
390 
391  if (udp_arg->srcport_start != udp_arg->srcport_end) {
392  udp->src_port = odp_cpu_to_be_16(udp_arg->srcport_crt);
393  if (udp_arg->srcport_crt >= udp_arg->srcport_end)
394  udp_arg->srcport_crt = udp_arg->srcport_start;
395  else
396  udp_arg->srcport_crt++;
397  }
398  if (udp_arg->dstport_start != udp_arg->dstport_end) {
399  udp->dst_port = odp_cpu_to_be_16(udp_arg->dstport_crt);
400  if (udp_arg->dstport_crt >= udp_arg->dstport_end)
401  udp_arg->dstport_crt = udp_arg->dstport_start;
402  else
403  udp_arg->dstport_crt++;
404  }
405 
406  udp->chksum = 0;
407  }
408 
409  if (pktout_cfg->bit.ipv4_chksum || pktout_cfg->bit.udp_chksum) {
410  odp_packet_l2_offset_set(pkt, 0);
411  odp_packet_l3_offset_set(pkt, ODPH_ETHHDR_LEN);
412  odp_packet_has_ipv4_set(pkt, 1);
413  odp_packet_l4_offset_set(pkt, ODPH_ETHHDR_LEN +
414  ODPH_IPV4HDR_LEN);
415  odp_packet_has_udp_set(pkt, 1);
416  }
417  return 0;
418 }

544 static int create_pktio(const char *dev, odp_pool_t pool,
545  uint32_t num_rx_queues,
546  uint32_t num_tx_queues,
547  interface_t *itf)
548 {
549  odp_pktio_capability_t capa;
550  int ret;
551  odp_pktio_param_t pktio_param;
552  odp_pktin_queue_param_t pktin_param;
553  odp_pktout_queue_param_t pktout_param;
554  odp_pktio_op_mode_t pktout_mode, pktin_mode;
555  odp_bool_t sched = args->appl.sched;
556 
557  odp_pktio_param_init(&pktio_param);
558  pktio_param.in_mode = num_rx_queues ?
559  (sched ? ODP_PKTIN_MODE_SCHED : ODP_PKTIN_MODE_DIRECT) :
560  ODP_PKTIN_MODE_DISABLED;
561  pktio_param.out_mode = num_tx_queues ? ODP_PKTOUT_MODE_DIRECT :
562  ODP_PKTOUT_MODE_DISABLED;
563 
564  /* Open a packet IO instance */
565  itf->pktio = odp_pktio_open(dev, pool, &pktio_param);
566 
567  if (itf->pktio == ODP_PKTIO_INVALID) {
568  ODPH_ERR("Error: pktio create failed for %s\n", dev);
569  return -1;
570  }
571 
572  if (odp_pktio_capability(itf->pktio, &capa)) {
573  ODPH_ERR("Error: Failed to get interface capabilities %s\n",
574  dev);
575  return -1;
576  }
577  odp_pktio_config_init(&itf->config);
578  if (args->appl.csum) {
579  itf->config.pktin.bit.ipv4_chksum =
580  capa.config.pktin.bit.ipv4_chksum;
581  itf->config.pktin.bit.udp_chksum =
582  capa.config.pktin.bit.udp_chksum;
583  itf->config.pktin.bit.drop_ipv4_err =
584  capa.config.pktin.bit.drop_ipv4_err;
585  itf->config.pktin.bit.drop_udp_err =
586  capa.config.pktin.bit.drop_udp_err;
587 
588  itf->config.pktout.bit.ipv4_chksum_ena =
589  capa.config.pktout.bit.ipv4_chksum_ena;
590  itf->config.pktout.bit.udp_chksum_ena =
591  capa.config.pktout.bit.udp_chksum_ena;
592  itf->config.pktout.bit.ipv4_chksum =
593  capa.config.pktout.bit.ipv4_chksum;
594  itf->config.pktout.bit.udp_chksum =
595  capa.config.pktout.bit.udp_chksum;
596  } else { /* explicit disable */
597  itf->config.pktin.bit.ipv4_chksum = 0;
598  itf->config.pktin.bit.udp_chksum = 0;
599  itf->config.pktout.bit.ipv4_chksum_ena = 0;
600  itf->config.pktout.bit.udp_chksum_ena = 0;
601  itf->config.pktout.bit.ipv4_chksum = 0;
602  itf->config.pktout.bit.udp_chksum = 0;
603  }
604 
605  itf->config.parser.layer = ODP_PROTO_LAYER_L2;
606  if (itf->config.pktin.bit.udp_chksum)
607  itf->config.parser.layer = ODP_PROTO_LAYER_L4;
608  else if (itf->config.pktin.bit.ipv4_chksum)
609  itf->config.parser.layer = ODP_PROTO_LAYER_L3;
610 
611  if (odp_pktio_config(itf->pktio, &itf->config)) {
612  ODPH_ERR("Error: Failed to set interface configuration %s\n",
613  dev);
614  return -1;
615  }
616 
617  if (num_rx_queues) {
618  pktin_mode = ODP_PKTIO_OP_MT_UNSAFE;
619  if (num_rx_queues > capa.max_input_queues) {
620  num_rx_queues = capa.max_input_queues;
621  pktin_mode = ODP_PKTIO_OP_MT;
622  ODPH_DBG("Warning: Force RX multithread safe mode "
623  "(slower)on %s\n", dev);
624  }
625 
626  odp_pktin_queue_param_init(&pktin_param);
627  pktin_param.num_queues = num_rx_queues;
628  pktin_param.op_mode = pktin_mode;
629  if (sched)
630  pktin_param.queue_param.sched.sync =
631  ODP_SCHED_SYNC_ATOMIC;
632 
633  if (odp_pktin_queue_config(itf->pktio, &pktin_param)) {
634  ODPH_ERR("Error: pktin queue config failed for %s\n",
635  dev);
636  return -1;
637  }
638  }
639 
640  if (num_tx_queues) {
641  pktout_mode = ODP_PKTIO_OP_MT_UNSAFE;
642  if (num_tx_queues > capa.max_output_queues) {
643  num_tx_queues = capa.max_output_queues;
644  pktout_mode = ODP_PKTIO_OP_MT;
645  ODPH_DBG("Warning: Force TX multithread safe mode "
646  "(slower) on %s\n", dev);
647  }
648 
649  odp_pktout_queue_param_init(&pktout_param);
650  pktout_param.num_queues = num_tx_queues;
651  pktout_param.op_mode = pktout_mode;
652 
653  if (odp_pktout_queue_config(itf->pktio, &pktout_param)) {
654  ODPH_ERR("Error: pktout queue config failed for %s\n",
655  dev);
656  return -1;
657  }
658  }
659 
660  ret = odp_pktio_start(itf->pktio);
661  if (ret)
662  ODPH_ABORT("Error: unable to start %s\n", dev);
663 
664  itf->pktout_count = num_tx_queues;
665  if (itf->pktout_count &&
666  odp_pktout_queue(itf->pktio, itf->pktout, itf->pktout_count) !=
667  (int)itf->pktout_count) {
668  ODPH_ERR("Error: failed to get output queues for %s\n", dev);
669  return -1;

701  odp_pktout_queue_t pktout;
702  odp_pktout_config_opt_t *pktout_cfg;

704  odp_packet_t pkt_array[MAX_UDP_TX_BURST];
705  int pkt_array_size, seq_step;
706  int burst_start, burst_size;
707  setup_pkt_ref_fn_t setup_pkt_ref = NULL;
708  setup_pkt_fn_t setup_pkt = NULL;
709  void *setup_pkt_arg = NULL;
710  counters_t *counters;
711  uint64_t pkt_count_max = 0;
712 
713  thr = odp_thread_id();
714  thr_args = arg;
715  pktout = thr_args->tx.pktout;
716  pktout_cfg = thr_args->tx.pktout_cfg;
717  counters = &thr_args->counters;
718 
719  /* Create reference packets*/
720  if (args->appl.mode == APPL_MODE_UDP) {
721  setup_pkt_ref = setup_udp_pkt_ref;
722  setup_pkt = setup_udp_pkt;
723  seq_step = args->tx_burst_size * (args->thread_cnt - 1);
724  if (args->appl.number != -1)
725  pkt_count_max = args->appl.number / args->thread_cnt +
726  (args->appl.number % args->thread_cnt ? 1 : 0);
727  setup_pkt_arg = &thr_args->tx.udp_param;
728  } else if (args->appl.mode == APPL_MODE_PING) {
729  setup_pkt_ref = setup_icmp_pkt_ref;
730  setup_pkt = setup_icmp_pkt;
731  seq_step = 0;
732  if (args->appl.number != -1)
733  pkt_count_max = args->appl.number;
734  } else {
735  ODPH_ERR(" [%02i] Error: invalid processing mode %d\n", thr,
736  args->appl.mode);
737  return -1;
738  }
739  pkt_array_size = args->tx_burst_size;
740 
741  if (setup_pkt_ref_array(thr_args->pool, pktout_cfg,
742  pkt_ref_array, pkt_array_size,
743  setup_pkt_ref)) {
744  ODPH_ERR("[%02i] Error: failed to create reference packets\n",
745  thr);
746  return -1;
747  }
748 
749  printf(" [%02i] created mode: SEND\n", thr);
750 
751  odp_barrier_wait(&args->barrier);
752 
753  for (;;) {
754  if (thr_args->stop)
755  break;
756 
757  if (pkt_count_max && counters->ctr_pkt_snd > pkt_count_max) {
758  sleep(1); /* wait for stop command */
759  continue;
760  }
761 
762  /* Setup TX burst*/
763  if (setup_pkt_array(pktout_cfg, counters,
764  pkt_ref_array, pkt_array,
765  pkt_array_size, setup_pkt, setup_pkt_arg)) {
766  ODPH_ERR("[%02i] Error: failed to setup packets\n",
767  thr);
768  break;
769  }
770 
771  /* Send TX burst*/
772  for (burst_start = 0, burst_size = pkt_array_size;;) {
773  ret = odp_pktout_send(pktout, &pkt_array[burst_start],
774  burst_size);
775  if (ret == burst_size) {
776  burst_size = 0;
777  break;
778  } else if (ret >= 0 && ret < burst_size) {
779  thr_args->counters.ctr_pkt_snd_drop +=
780  burst_size - ret;
781 
782  burst_start += ret;
783  burst_size -= ret;
784  continue;
785  }
786  ODPH_ERR(" [%02i] packet send failed\n", thr);
787  odp_packet_free_multi(&pkt_array[burst_start],
788  burst_size);
789  break;
790  }
791 
792  counters->ctr_pkt_snd += pkt_array_size - burst_size;
793 
794  if (args->appl.interval != 0)
795  odp_time_wait_ns((uint64_t)args->appl.interval *
796  ODP_TIME_MSEC_IN_NS);
797  counters->ctr_seq += seq_step;
798  }
799 
800  odp_packet_free_multi(pkt_ref_array, pkt_array_size);
801 
802  return 0;

847 static void process_pkts(int thr, thread_args_t *thr_args,
848  odp_packet_t pkt_tbl[], unsigned len)
849 {
850  odp_packet_t pkt;
851  uint32_t left, offset, i;
852  odph_ipv4hdr_t *ip;
853 
854  for (i = 0; i < len; ++i) {
855  pkt = pkt_tbl[i];
856 
857  /* Drop packets with errors */
858  if (odp_unlikely(odp_packet_has_error(pkt)))
859  continue;
860 
861  offset = odp_packet_l3_offset(pkt);
862  left = odp_packet_len(pkt) - offset;
863 
864  if (left < sizeof(odph_ipv4hdr_t))
865  continue;
866 
867  ip = (odph_ipv4hdr_t *)((uint8_t *)odp_packet_data(pkt) +
868  offset);
869 
870  /* only ip pkts */
871  if (ODPH_IPV4HDR_VER(ip->ver_ihl) != ODPH_IPV4)
872  continue;
873 
874  thr_args->counters.ctr_pkt_rcv++;
875 
876  /* udp */
877  if (ip->proto == ODPH_IPPROTO_UDP) {
878  thr_args->counters.ctr_udp_rcv++;
879  } else if (ip->proto == ODPH_IPPROTO_ICMPV4) {
880  uint32_t l3_size = ODPH_IPV4HDR_IHL(ip->ver_ihl) * 4;
881 
882  offset += l3_size;
883  left -= l3_size;
884 
885  if (left < sizeof(odph_icmphdr_t))
886  continue;
887 
888  process_icmp_pkt(thr, thr_args,
889  (uint8_t *)odp_packet_data(pkt) +
890  offset);
891  }
892  }

1020 static void print_global_stats(int num_workers)
1021 {
1022  odp_time_t cur, wait, next, left;
1023  uint64_t pkts_snd = 0, pkts_snd_prev = 0;
1024  uint64_t pps_snd = 0, maximum_pps_snd = 0;
1025  uint64_t pkts_rcv = 0, pkts_rcv_prev = 0;
1026  uint64_t pps_rcv = 0, maximum_pps_rcv = 0;
1027  uint64_t stall, pkts_snd_drop;
1028  int verbose_interval = STATS_INTERVAL, i;
1029  odp_thrmask_t thrd_mask;
1030 
1031  odp_barrier_wait(&args->barrier);
1032 
1033  wait = odp_time_local_from_ns(verbose_interval * ODP_TIME_SEC_IN_NS);
1034  next = odp_time_sum(odp_time_local(), wait);
1035 
1036  while (odp_thrmask_worker(&thrd_mask) == num_workers) {
1037  if (args->appl.mode != APPL_MODE_RCV &&
1038  args->appl.number != -1) {
1039  uint64_t cnt = COUNTER_SUM(pkt_snd, num_workers);
1040 
1041  if (cnt >= (unsigned int)args->appl.number) {
1042  garceful_stop_ping();
1043  break;
1044  }
1045  }
1046  cur = odp_time_local();
1047  if (odp_time_cmp(next, cur) > 0) {
1048  left = odp_time_diff(next, cur);
1049  stall = odp_time_to_ns(left);
1050  if (stall / ODP_TIME_SEC_IN_NS)
1051  sleep(1);
1052  else
1053  usleep(stall / ODP_TIME_USEC_IN_NS);
1054  continue;
1055  }
1056  next = odp_time_sum(cur, wait);
1057 
1058  switch (args->appl.mode) {
1059  case APPL_MODE_RCV:
1060  pkts_rcv = COUNTER_SUM(pkt_rcv, num_workers);
1061  pkts_snd = 0;
1062  pkts_snd_drop = 0;
1063  break;
1064  case APPL_MODE_PING:
1065  pkts_snd = COUNTER_SUM(pkt_snd, num_workers);
1066  pkts_snd_drop = COUNTER_SUM(pkt_snd_drop, num_workers);
1067  pkts_rcv = COUNTER_SUM(icmp_reply_rcv, num_workers);
1068  break;
1069  case APPL_MODE_UDP:
1070  pkts_snd = COUNTER_SUM(pkt_snd, num_workers);
1071  pkts_snd_drop = COUNTER_SUM(pkt_snd_drop, num_workers);
1072  break;
1073  default:
1074  continue;
1075  }
1076 
1077  pps_snd = (pkts_snd - pkts_snd_prev) / verbose_interval;
1078  pkts_snd_prev = pkts_snd;
1079  if (pps_snd > maximum_pps_snd)
1080  maximum_pps_snd = pps_snd;
1081 
1082  pps_rcv = (pkts_rcv - pkts_rcv_prev) / verbose_interval;
1083  pkts_rcv_prev = pkts_rcv;
1084  if (pps_rcv > maximum_pps_rcv)
1085  maximum_pps_rcv = pps_rcv;
1086 
1087  printf("sent: %" PRIu64 ", drops: %" PRIu64 ", "
1088  "send rate: %" PRIu64 " pps, "
1089  "max send rate: %" PRIu64 " pps, "
1090  "rcv: %" PRIu64 ", "
1091  "recv rate: %" PRIu64 " pps, "
1092  "max recv rate: %" PRIu64 " pps\n",
1093  pkts_snd, pkts_snd_drop,
1094  pps_snd, maximum_pps_snd,
1095  pkts_rcv, pps_rcv, maximum_pps_rcv);
1096  fflush(NULL);
1097  }
1098 
1099  for (i = 0; i < num_workers; i++)
1100  args->thread[i].stop = 1;
1101 }
