88static void time_elapsed(odp_time_t *t, int thr)
89{
90 odp_time_t stop = odp_time_local_strict();
91 uint64_t nsec = odp_time_diff_ns(stop, *t);
92
93 *t = stop;
94 glb->stat[thr].sum += nsec;
95 if (nsec < glb->stat[thr].min || !glb->stat[thr].min)
96 glb->stat[thr].min = nsec;
97 if (nsec > glb->stat[thr].max)
98 glb->stat[thr].max = nsec;
99}

101static void *read_file(const char *name, uint64_t *size)
102{
103 void *addr = NULL;
104 FILE *file = fopen(name, "rb");
105
106 if (!file) {
107 ODPH_ERR("Failed to open file %s: %s\n", name, strerror(errno));
108 return NULL;
109 }
110
111 if (fseek(file, 0, SEEK_END)) {
112 ODPH_ERR("Failed to get file size for file %s\n", name);
113 goto error;
114 }
115
116 long pos = ftell(file);
117
118 if (pos < 0) {
119 ODPH_ERR("Failed to get file size for file %s\n", name);
120 goto error;
121 }
122
123 rewind(file);
124 *size = pos;
125 addr = malloc(*size);
126
127 if (!addr) {
128 ODPH_ERR("Allocating %" PRIu64 " bytes failed\n", *size);
129 goto error;
130 }
131
132 if (fread(addr, *size, 1, file) != 1) {
133 ODPH_ERR("Reading %" PRIu64 " bytes failed\n", *size);
134 goto error;
135 }
136
137 fclose(file);
138 printf("Read %" PRIu64 " bytes from %s\n", *size, name);
139
140 return addr;
141
142error:
143 fclose(file);
144 free(addr);
145
146 return NULL;
147}

149static int read_file_to_shm(odp_shm_t *shm, void **data, uint64_t *size, const char *filename,
150 const char *name)
151{
152 void *file = read_file(filename, size);
153
154 if (!file)
155 return -1;
156
157 *shm = odp_shm_reserve(name, *size, ODP_CACHE_LINE_SIZE, 0);
158
159 if (*shm != ODP_SHM_INVALID)
160 *data = odp_shm_addr(*shm);
161
162 if (!*data) {
163 ODPH_ERR("Failed to reserve shm for %s\n", filename);
164 free(file);
165 return -1;
166 }
167
168 memcpy(*data, file, *size);
169 free(file);
170
171 return 0;
172}

203static int parse_args(int argc, char *argv[])
204{
205 static const struct option longopts[] = {
206 { "model", required_argument, NULL, 'M' },
207 { "input", required_argument, NULL, 'I' },
208 { "num_cpu", required_argument, NULL, 'c' },
209 { "rounds", required_argument, NULL, 'r' },
210 { "mode", required_argument, NULL, 'm' },
211 { "latency", no_argument, NULL, 'l' },
212 { "warmup", required_argument, NULL, 'w' },
213 { "reference", required_argument, NULL, 'R' },
214 { "quant", required_argument, NULL, 'q' },
215 { "dequant", required_argument, NULL, 'd' },
216 { "batches", required_argument, NULL, 'b' },
217 { "help", no_argument, NULL, 'h' },
218 { NULL, 0, NULL, 0 } };
219
220 static const char *shortopts = "+M:I:c:r:m:lw:R:q:d:b:h";
221
222 glb->opt = opt_def;
223
224 while (1) {
225 int c = getopt_long(argc, argv, shortopts, longopts, NULL);
226
227 if (c == -1)
228 break; /* No more options */
229
230 switch (c) {
231 case 'M':
232 glb->opt.model_name = optarg;
233 break;
234 case 'I':
235 glb->opt.input_name = optarg;
236 break;
237 case 'c':
238 glb->opt.num_threads = atoi(optarg);
239 break;
240 case 'r':
241 glb->opt.rounds = atoi(optarg);
242 break;
243 case 'm':
244 glb->opt.mode = atoi(optarg);
245 break;
246 case 'l':
247 glb->opt.latency = 1;
248 break;
249 case 'w':
250 glb->opt.warmup = atoi(optarg);
251 break;
252 case 'R':
253 glb->opt.reference_name = optarg;
254 break;
255 case 'q':
256 glb->opt.scale_q = atof(optarg);
257 break;
258 case 'd':
259 glb->opt.scale_d = atof(optarg);
260 break;
261 case 'b':
262 glb->opt.num_batch = atoi(optarg);
263 break;
264 case 'h':
265 usage(argv[0]);
266 return 1;
267 default:
268 usage(argv[0]);
269 return -1;
270 }
271 }
272
273 optind = 1; /* reset 'extern optind' from the getopt lib */
274
275 if (!glb->opt.model_name || !glb->opt.input_name) {
276 ODPH_ERR("Model and input files are mandatory\n");
277 exit(EXIT_FAILURE);
278 }
279
280 if (glb->opt.mode < 0 || glb->opt.mode >= MODE_NUM) {
281 ODPH_ERR("Invalid mode %d\n", glb->opt.mode);
282 exit(EXIT_FAILURE);
283 }
284
285 printf("Options:\n");
286 printf("--------\n");
287 printf("model_name: %s\n", glb->opt.model_name);
288 printf("input_name: %s\n", glb->opt.input_name);
289 printf("num_cpu: %d\n", glb->opt.num_threads);
290 printf("rounds: %u\n", glb->opt.rounds);
291 printf("mode: %u\n", glb->opt.mode);
292 printf("latency: %d\n", glb->opt.latency);
293 printf("warmup: %d\n", glb->opt.warmup);
294 printf("reference_name: %s\n", glb->opt.reference_name);
295 printf("scale_q: %g\n", glb->opt.scale_q);
296 printf("scale_d: %g\n", glb->opt.scale_d);
297 printf("num_batch: %d\n", glb->opt.num_batch);
298 printf("\n");
299
300 return 0;
301}

303static int check_num_batch(void)
304{
305 int min_batch = 1, max_batch = 1;
306
307 for (int i = 0; i < glb->num_inp; i++) {
308 odp_ml_shape_info_t *shape = &glb->inp_info[i].shape;
309
310 for (int j = 0; j < (int)shape->num_dim; j++) {
311 if (shape->dim[j] == ODP_ML_DIM_DYNAMIC) {
312 min_batch = shape->dim_min[j];
313 max_batch = shape->dim_max[j];
314 break;
315 }
316 }
317 }
318
319 if (glb->opt.num_batch < min_batch || glb->opt.num_batch > max_batch) {
320 ODPH_ERR("Number of batches %d out of range [%d, %d]\n", glb->opt.num_batch,
321 min_batch, max_batch);
322 return -1;
323 }
324
325 return 0;
326}

328static void calc_io_size(void)
329{
330 for (int i = 0; i < glb->num_inp; i++) {
331 uint64_t elems = 1;
332 odp_ml_input_info_t *info = &glb->inp_info[i];
333 odp_ml_shape_info_t *shape = &info->shape;
334 io_size *inp = &glb->inp[i];
335
336 printf("Input %d: %s, shape:", i, info->name);
337
338 for (int j = 0; j < (int)shape->num_dim; j++) {
339 printf(" %d", shape->dim[j]);
340 if (shape->dim[j] != ODP_ML_DIM_DYNAMIC)
341 elems *= shape->dim[j];
342 }
343
344 if (shape->type == ODP_ML_SHAPE_BATCH)
345 elems *= glb->opt.num_batch;
346 inp->elems = elems;
347 inp->elem_size = info->data_type_size;
348 inp->size = elems * info->data_type_size;
349 glb->inp_size_q += inp->size;
350 glb->inp_size_d += elems * sizeof(float);
351
352 printf(", elems: %" PRIu64 ", datatype size: %d, size: %" PRIu64 "\n", inp->elems,
353 inp->elem_size, inp->size);
354 }
355
356 printf("Input size_q: %" PRIu64 ", size_d: %" PRIu64 "\n", glb->inp_size_q,
357 glb->inp_size_d);
358
359 for (int i = 0; i < glb->num_out; i++) {
360 uint64_t elems = 1;
361 odp_ml_output_info_t *info = &glb->out_info[i];
362 odp_ml_shape_info_t *shape = &info->shape;
363 io_size *out = &glb->out[i];

365 printf("Output %d: %s, shape:", i, info->name);
366
367 for (int j = 0; j < (int)shape->num_dim; j++) {
368 printf(" %d", shape->dim[j]);
369 if (shape->dim[j] != ODP_ML_DIM_DYNAMIC)
370 elems *= shape->dim[j];

372
373 if (shape->type == ODP_ML_SHAPE_BATCH)

381 printf(", elems: %" PRIu64 ", datatype size: %d, size: %" PRIu64 "\n", out->elems,

385 printf("Output size_q: %" PRIu64 ", size_d: %" PRIu64 "\n", glb->out_size_q,

401static void quantize_input(uint8_t *inp_q_addr, uint8_t *inp_d_addr)
402{
403 for (int i = 0; i < glb->num_inp; i++) {
404 float scale_q = glb->opt.scale_q;
405 uint64_t elems = glb->inp[i].elems;
406 odp_ml_data_type_t data_type = glb->inp_info[i].data_type;
407
408 switch (data_type) {
409 case ODP_ML_DATA_TYPE_INT8:
410 odp_ml_fp32_to_int8((int8_t *)inp_q_addr, (float *)inp_d_addr, elems,
411 scale_q, 0);
412 break;
413 case ODP_ML_DATA_TYPE_UINT8:
414 odp_ml_fp32_to_uint8((uint8_t *)inp_q_addr, (float *)inp_d_addr, elems,
415 scale_q, 0);
416 break;
417 case ODP_ML_DATA_TYPE_FP16:
418 odp_ml_fp32_to_fp16((uint16_t *)inp_q_addr, (float *)inp_d_addr, elems);
419 break;
420 default:
421 ODPH_ERR("Unsupported type %d for input %d\n", data_type, i);
422 }
423
424 inp_q_addr += glb->inp[i].size;
425 inp_d_addr += elems * sizeof(float);
426 }
427}

445 case ODP_ML_DATA_TYPE_FP16:

451
452 out_q_addr += glb->out[i].size;
453 out_d_addr += elems * sizeof(float);
454 }

457static int test_ml(void *ptr)
458{
459 odp_time_t time;
460 odp_ml_data_seg_t inp_seg[MAX_IO];
461 uint8_t *inp_addr;
462 odp_ml_data_seg_t out_seg[MAX_IO];
463 uint8_t *out_addr;
464 int thread_idx = (int)(uintptr_t)ptr;
465 void *input = NULL, *output = NULL, *output_file = NULL;
466 void *output_final = NULL;
467 uint64_t out_size_final = 0;
468 int ret = 0;
469
470 inp_addr = glb->inp_file_data;
471
472 if (glb->opt.scale_q > 0.0) {
473 input = malloc(glb->inp_size_q);
474 if (!input) {
475 ODPH_ERR("Allocating %" PRIu64 " bytes failed\n", glb->inp_size_q);
476 ret = -1;
477 goto error;
478 }
479
480 inp_addr = input;
481 }
482
483 for (int i = 0; i < glb->num_inp; i++) {
484 inp_seg[i].addr = inp_addr;
485 inp_seg[i].size = glb->inp[i].size;
486 inp_addr += glb->inp[i].size;
487 }
488
489 output = malloc(glb->out_size_q);
490
491 if (!output) {
492 ODPH_ERR("Allocating %" PRIu64 " bytes failed\n", glb->out_size_q);
493 ret = -1;
494 goto error;
495 }
496
497 out_addr = output;
498 output_final = output;
499 out_size_final = glb->out_size_q;
500
501 if (glb->opt.scale_d > 0.0) {
502 output_file = malloc(glb->out_size_d);
503 if (!output_file) {
504 ODPH_ERR("Allocating %" PRIu64 " bytes failed\n", glb->out_size_d);
505 ret = -1;
506 goto error;
507 }
508
509 output_final = output_file;
510 out_size_final = glb->out_size_d;
511 }
512
513 for (int i = 0; i < glb->num_out; i++) {
514 out_seg[i].addr = out_addr;
515 out_seg[i].size = glb->out[i].size;
516 out_addr += glb->out[i].size;
517 }
518
519 odp_ml_data_t data = {
520 .input_seg = inp_seg,
521 .num_input_seg = glb->num_inp,
522 .output_seg = out_seg,
523 .num_output_seg = glb->num_out,
524 };
525 odp_ml_run_param_t run_param;
526
527 odp_ml_run_param_init(&run_param);
528 run_param.batch_size = glb->opt.num_batch;
529
530 if (glb->opt.mode != MODE_INFERENCE_QUANT && glb->opt.scale_q > 0.0)
531 quantize_input(input, glb->inp_file_data);
532
533 odp_barrier_wait(&glb->barrier);
534
535 for (int i = 0; i < glb->opt.rounds + glb->opt.warmup; i++) {
536 if (i == glb->opt.warmup)
537 time_start(&time);
538
539 if (glb->opt.mode == MODE_INFERENCE_QUANT && glb->opt.scale_q > 0.0)
540 quantize_input(input, glb->inp_file_data);
541
542 int r;
543
544 while (!(r = odp_ml_run(glb->mdl, &data, &run_param)))
545 ;
546 if (r != 1) {
547 ODPH_ERR("odp_ml_run() failed\n");
548 ret = -1;
549 goto error;
550 }
551
552 if (glb->opt.mode == MODE_INFERENCE_QUANT && glb->opt.scale_d > 0.0)
553 dequantize_output(output_file, output);
554
555 if (glb->opt.latency && i >= glb->opt.warmup)
556 time_elapsed(&time, thread_idx);
557 }
558
559 if (!glb->opt.latency)
560 time_elapsed(&time, thread_idx);
561
562 if (glb->opt.mode != MODE_INFERENCE_QUANT && glb->opt.scale_d > 0.0)
563 dequantize_output(output_file, output);
564
565 if (glb->ref_file_data) {
566 if (out_size_final != glb->ref_file_size) {
567 ODPH_ERR("Output size mismatch: %" PRIu64
568 " differs from reference file size %" PRIu64 "\n",
569 out_size_final, glb->ref_file_size);
570 ret = -1;
571 goto error;
572 }
573
574 if (memcmp(glb->ref_file_data, output_final, out_size_final)) {
575 ODPH_ERR("Output differs from reference\n");
576 ret = -1;
577 goto error;
578 }
579 }
580
581error:
582 free(input);
583 free(output);
584 free(output_file);
585
586 return ret;
587}

685static void print_results_avg(void)
686{
687 int num_threads = glb->opt.num_threads;
688 int rounds = glb->opt.rounds;
689
690 printf("thread %15s %15s\n", "avg (nsec)", "rounds / sec");
691 printf("--------------------------------------\n");
692 if (num_threads > 1) {
693 uint64_t avg = 0;
694
695 for (int i = 0; i < num_threads; i++)
696 avg += glb->stat[i].sum;
697
698 avg /= rounds * num_threads;
699 printf("%-6s %15" PRIu64 " %15.1f\n", "all", avg, (float)ODP_TIME_SEC_IN_NS / avg);
700 }
701 for (int i = 0; i < num_threads; i++) {
702 glb->stat[i].sum /= rounds;
703 printf("%-6d %15" PRIu64 " %15.1f\n", i, glb->stat[i].sum,
704 (float)ODP_TIME_SEC_IN_NS / glb->stat[i].sum);
705 }
706}

708static void print_stat(stat_t *stat)
709{
710 printf("%15" PRIu64 " %15" PRIu64 " %15" PRIu64, stat->min, stat->sum, stat->max);
711}
712
713static void print_results_min_avg_max(void)
714{
715 int num_threads = glb->opt.num_threads;
716 int rounds = glb->opt.rounds;
717
718 printf("thread %15s %15s %15s\n", "min (nsec)", "avg (nsec)", "max (nsec)");
719 printf("------------------------------------------------------\n");
720 if (num_threads > 1) {
721 stat_t s = { 0 };
722
723 for (int i = 0; i < num_threads; i++) {
724 s.sum += glb->stat[i].sum;
725 if (glb->stat[i].min && (glb->stat[i].min < s.min || !s.min))
726 s.min = glb->stat[i].min;
727 if (glb->stat[i].max > s.max)
728 s.max = glb->stat[i].max;
729 }
730
731 s.sum /= rounds * num_threads;
732 printf("%-6s ", "all");
733 print_stat(&s);
734 printf("\n");
735 }
736 for (int i = 0; i < num_threads; i++) {
737 glb->stat[i].sum /= rounds;
738 printf("%-6d ", i);
739 print_stat(&glb->stat[i]);
740 printf("\n");
741 }
742}

745{
746 printf("\n");
747
748 if (glb->opt.latency)
749 print_results_min_avg_max();
750 else
751 print_results_avg();
752
753 printf("\n");
