42_ODP_INLINE void odp_rwlock_read_lock(odp_rwlock_t *rwlock)
43{
44 uint32_t cnt;
45 int is_locked = 0;
46
47 while (is_locked == 0) {
48 cnt = odp_atomic_load_u32(&rwlock->cnt);
49 /* waiting for read lock */
50 if ((int32_t)cnt < 0) {
51 odp_cpu_pause();
52 continue;
53 }
54 is_locked = odp_atomic_cas_acq_u32(&rwlock->cnt, &cnt, cnt + 1);
55 }
56}
57
58_ODP_INLINE int odp_rwlock_read_trylock(odp_rwlock_t *rwlock)
59{
60 uint32_t cnt = odp_atomic_load_u32(&rwlock->cnt);
61
62 while (cnt != (uint32_t)-1) {
63 if (odp_atomic_cas_acq_u32(&rwlock->cnt, &cnt, cnt + 1))
64 return 1;
65 }
66
67 return 0;
68}

70_ODP_INLINE void odp_rwlock_read_unlock(odp_rwlock_t *rwlock)
71{
72 odp_atomic_sub_rel_u32(&rwlock->cnt, 1);
73}

75_ODP_INLINE void odp_rwlock_write_lock(odp_rwlock_t *rwlock)
76{
77 uint32_t cnt;
78 int is_locked = 0;
79
80 while (is_locked == 0) {
81 uint32_t zero = 0;
82
83 cnt = odp_atomic_load_u32(&rwlock->cnt);
84 /* lock acquired, wait */
85 if (cnt != 0) {
86 odp_cpu_pause();
87 continue;
88 }
89 is_locked = odp_atomic_cas_acq_u32(&rwlock->cnt, &zero, (uint32_t)-1);
90 }
91}
92
93_ODP_INLINE int odp_rwlock_write_trylock(odp_rwlock_t *rwlock)
94{
95 uint32_t zero = 0;
96
97 return odp_atomic_cas_acq_u32(&rwlock->cnt, &zero, (uint32_t)-1);
98}

100_ODP_INLINE void odp_rwlock_write_unlock(odp_rwlock_t *rwlock)
101{
102 odp_atomic_store_rel_u32(&rwlock->cnt, 0);
103}
