70_ODP_INLINE uint64_t odp_cpu_cycles_diff(uint64_t c2, uint64_t c1)
71{
72 if (odp_likely(c2 >= c1))
73 return c2 - c1;
74
75 return c2 + (odp_cpu_cycles_max() - c1) + _odp_cpu_cycles_resolution();
76}

78_ODP_INLINE void odp_prefetch(const void *ptr)
79{
80 /*
81 * __builtin_prefetch (const void *addr, rw, locality)
82 *
83 * rw 0..1 (0: read, 1: write)
84 * locality 0..3 (0: don't leave to cache, 3: leave on all cache levels)
85 */
86 __builtin_prefetch(ptr, 0, 3);
87}
