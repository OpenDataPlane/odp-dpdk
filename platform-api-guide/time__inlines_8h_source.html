49 static inline uint64_t _odp_time_to_ns(odp_time_t time)
50 {
51  uint64_t nsec;
52  uint64_t count = time.count;
53  uint64_t sec = 0;
54  const uint64_t freq_hz = _odp_time_glob.freq_hz;
55  const uint64_t giga_hz = 1000000000;
56 
57  if (count >= freq_hz) {
58  sec = count / freq_hz;
59  count = count - sec * freq_hz;
60  }
61 
62  nsec = (giga_hz * count) / freq_hz;
63 
64  return (sec * giga_hz) + nsec;
65 }

67 static inline odp_time_t _odp_time_from_ns(uint64_t ns)
68 {
69  odp_time_t time;
70  uint64_t count;
71  uint64_t sec = 0;
72  const uint64_t freq_hz = _odp_time_glob.freq_hz;
73 
74  if (ns >= ODP_TIME_SEC_IN_NS) {
75  sec = ns / ODP_TIME_SEC_IN_NS;
76  ns = ns - sec * ODP_TIME_SEC_IN_NS;
77  }
78 
79  count = sec * freq_hz;
80  count += (ns * freq_hz) / ODP_TIME_SEC_IN_NS;
81 
82  time.count = count;
83 
84  return time;
85 }
