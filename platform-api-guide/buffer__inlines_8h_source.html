101 _ODP_INLINE void odp_buffer_free_multi(const odp_buffer_t buf[], int num)
102 {
103  struct rte_mbuf *mbuf_tbl[num];
104  struct rte_mempool *mp_pending;
105  unsigned int num_pending;
106 
107  if (odp_unlikely(num <= 0))
108  return;
109 
110  _odp_buffer_validate_multi(buf, num, _ODP_EV_BUFFER_FREE_MULTI);
111 
112  mbuf_tbl[0] = (struct rte_mbuf *)buf[0];
113  mp_pending = mbuf_tbl[0]->pool;
114  num_pending = 1;
115 
116 /*
117  * num_pending is less than or equal to num, but GCC 13 is not able figure that out, so we have to
118  * ignore array-bounds warnings in the rte_mempool_put_bulk() calls.
119  */
120 #pragma GCC diagnostic push
121 #pragma GCC diagnostic ignored "-Warray-bounds"
122  for (int i = 1; i < num; i++) {
123  struct rte_mbuf *mbuf = (struct rte_mbuf *)buf[i];
124 
125  if (mbuf->pool != mp_pending) {
126  rte_mempool_put_bulk(mp_pending, (void **)mbuf_tbl, num_pending);
127  mbuf_tbl[0] = mbuf;
128  num_pending = 1;
129  mp_pending = mbuf->pool;
130  } else {
131  mbuf_tbl[num_pending++] = mbuf;
132  }
133  }
134  rte_mempool_put_bulk(mp_pending, (void **)mbuf_tbl, num_pending);
135 #pragma GCC diagnostic pop
136 }

138 _ODP_INLINE int odp_buffer_is_valid(odp_buffer_t buf)
139 {
140  if (odp_event_is_valid(odp_buffer_to_event(buf)) == 0)
141  return 0;
142 
143  if (odp_event_type(odp_buffer_to_event(buf)) != ODP_EVENT_BUFFER)
144  return 0;
145 
146  if (odp_unlikely(_odp_buffer_validate(buf, _ODP_EV_BUFFER_IS_VALID)))
147  return 0;
148 
149  return 1;
150 }
