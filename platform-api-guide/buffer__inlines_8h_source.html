97_ODP_INLINE void odp_buffer_free(odp_buffer_t buf)
98{
99 struct rte_mbuf *mbuf = (struct rte_mbuf *)buf;
100
101 _odp_buffer_validate(buf, _ODP_EV_BUFFER_FREE);
102
103 rte_mempool_put(mbuf->pool, mbuf);
104}

106_ODP_INLINE void odp_buffer_free_multi(const odp_buffer_t buf[], int num)
107{
108 struct rte_mempool *mp_pending;
109 struct rte_mbuf *mbuf;
110 uint32_t i, num_pending;
111
112 if (odp_unlikely(num <= 0))
113 return;
114
115 _odp_buffer_validate_multi(buf, num, _ODP_EV_BUFFER_FREE_MULTI);
116
117 mbuf = (struct rte_mbuf *)buf[0];
118 mp_pending = mbuf->pool;
119 num_pending = 1;
120
121 for (i = 1; i < (uint32_t)num; i++) {
122 mbuf = (struct rte_mbuf *)buf[i];
123
124 if (mbuf->pool != mp_pending) {
125 rte_mempool_put_bulk(mp_pending, (void **)(uintptr_t)&buf[i - num_pending],
126 num_pending);
127 mp_pending = mbuf->pool;
128 num_pending = 0;
129 }
130 num_pending++;
131 }
132 rte_mempool_put_bulk(mp_pending, (void **)(uintptr_t)&buf[i - num_pending], num_pending);
133}

135_ODP_INLINE int odp_buffer_is_valid(odp_buffer_t buf)
136{
137 if (odp_event_is_valid(odp_buffer_to_event(buf)) == 0)
138 return 0;
139
140 if (odp_event_type(odp_buffer_to_event(buf)) != ODP_EVENT_BUFFER)
141 return 0;
142
143 if (odp_unlikely(_odp_buffer_validate(buf, _ODP_EV_BUFFER_IS_VALID)))
144 return 0;
145
146 return 1;
147}
