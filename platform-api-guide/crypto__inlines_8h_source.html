44_ODP_INLINE int odp_crypto_result(odp_crypto_packet_result_t *result, odp_packet_t pkt)
45{
46 odp_crypto_packet_result_t *op_result;
47 odp_bool_t ok;
48
49 _ODP_ASSERT(odp_packet_subtype(pkt) == ODP_EVENT_PACKET_CRYPTO);
50
51 op_result = _odp_pkt_get_ptr(pkt, odp_crypto_packet_result_t, crypto_op);
52
53 ok = op_result->cipher_status.alg_err == ODP_CRYPTO_ALG_ERR_NONE &&
54 op_result->auth_status.alg_err == ODP_CRYPTO_ALG_ERR_NONE;
55
56 if (result)
57 *result = *op_result;
58
59 return ok ? 0 : -1;
60}
