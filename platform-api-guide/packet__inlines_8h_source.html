133_ODP_INLINE void *odp_packet_offset(odp_packet_t pkt, uint32_t offset,
134 uint32_t *len, odp_packet_seg_t *seg)
135{
136 struct rte_mbuf *mb = (struct rte_mbuf *)pkt;
137
138 if (odp_unlikely(offset == ODP_PACKET_OFFSET_INVALID))
139 goto err;
140
141 do {
142 if (mb->data_len > offset)
143 break;
144 offset -= mb->data_len;
145 mb = mb->next;
146 } while (mb);
147
148 if (mb) {
149 if (len)
150 *len = mb->data_len - offset;
151 if (seg)
152 *seg = (odp_packet_seg_t)(uintptr_t)mb;
153 return (void *)(rte_pktmbuf_mtod(mb, char *) + offset);
154 }
155err:
156 if (len)
157 *len = 0;
158 if (seg)
159 *seg = NULL;
160 return NULL;
161}

391 _odp_packet_input_flags_t input_flags;
392
393 input_flags.all = _odp_pkt_get(pkt, uint64_t, input_flags);
394
395 if (input_flags.tcp)
396 return ODP_PROTO_L4_TYPE_TCP;
397 else if (input_flags.udp)

399 else if (input_flags.sctp)
400 return ODP_PROTO_L4_TYPE_SCTP;
401 else if (input_flags.ipsec_ah)
402 return ODP_PROTO_L4_TYPE_AH;
403 else if (input_flags.ipsec_esp)
404 return ODP_PROTO_L4_TYPE_ESP;
405 else if (input_flags.icmp && input_flags.ipv4)
406 return ODP_PROTO_L4_TYPE_ICMPV4;
407 else if (input_flags.icmp && input_flags.ipv6)
408 return ODP_PROTO_L4_TYPE_ICMPV6;
409 else if (input_flags.no_next_hdr)
410 return ODP_PROTO_L4_TYPE_NO_NEXT;
411
412 return ODP_PROTO_L4_TYPE_NONE;

536_ODP_INLINE void odp_packet_prefetch(odp_packet_t pkt, uint32_t offset,
537 uint32_t len)
538{
539 const char *addr = (char *)odp_packet_data(pkt) + offset;
540 size_t ofs;
541
542 for (ofs = 0; ofs < len; ofs += RTE_CACHE_LINE_SIZE)
543 rte_prefetch0(addr + ofs);
544}

546static inline int _odp_packet_copy_to_mem_seg(odp_packet_t pkt, uint32_t offset,
547 uint32_t len, void *dst)
548{
549 void *mapaddr;
550 uint32_t seglen = 0; /* GCC */
551 uint32_t cpylen;
552 uint8_t *dstaddr = (uint8_t *)dst;
553
554 if (offset + len > odp_packet_len(pkt))
555 return -1;
556
557 while (len > 0) {
558 mapaddr = odp_packet_offset(pkt, offset, &seglen, NULL);
559 cpylen = len > seglen ? seglen : len;
560 rte_memcpy(dstaddr, mapaddr, cpylen);
561 offset += cpylen;
562 dstaddr += cpylen;
563 len -= cpylen;
564 }
565
566 return 0;
567}

583static inline int _odp_packet_copy_from_mem_seg(odp_packet_t pkt,
584 uint32_t offset, uint32_t len,
585 const void *src)
586{
587 void *mapaddr;
588 uint32_t seglen = 0; /* GCC */
589 uint32_t cpylen;
590 const uint8_t *srcaddr = (const uint8_t *)src;
591
592 if (offset + len > odp_packet_len(pkt))
593 return -1;
594
595 while (len > 0) {
596 mapaddr = odp_packet_offset(pkt, offset, &seglen, NULL);
597 cpylen = len > seglen ? seglen : len;
598 rte_memcpy(mapaddr, srcaddr, cpylen);
599 offset += cpylen;
600 srcaddr += cpylen;
601 len -= cpylen;
602 }
603
604 return 0;
605}
