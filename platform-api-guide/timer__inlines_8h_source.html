90_ODP_INLINE uint64_t odp_timer_tick_to_ns(odp_timer_pool_t tp ODP_UNUSED, uint64_t ticks)
91{
92 uint64_t nsec;
93 uint64_t sec = 0;
94 const uint64_t freq_hz = _odp_timer_glob.freq_hz;
95
96 if (ticks >= freq_hz) {
97 sec = ticks / freq_hz;
98 ticks = ticks - sec * freq_hz;
99 }
100
101 nsec = (ODP_TIME_SEC_IN_NS * ticks) / freq_hz;
102
103 return (sec * ODP_TIME_SEC_IN_NS) + nsec;
104}
105
106_ODP_INLINE uint64_t odp_timer_ns_to_tick(odp_timer_pool_t tp ODP_UNUSED, uint64_t ns)
107{
108 uint64_t ticks;
109 uint64_t sec = 0;
110 const uint64_t freq_hz = _odp_timer_glob.freq_hz;
111
112 if (ns >= ODP_TIME_SEC_IN_NS) {
113 sec = ns / ODP_TIME_SEC_IN_NS;
114 ns = ns - sec * ODP_TIME_SEC_IN_NS;
115 }
116
117 ticks = sec * freq_hz;
118 ticks += (ns * freq_hz) / ODP_TIME_SEC_IN_NS;
119
120 return ticks;
121}
