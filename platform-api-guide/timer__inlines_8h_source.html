86 _ODP_INLINE uint64_t odp_timer_tick_to_ns(odp_timer_pool_t tp ODP_UNUSED, uint64_t ticks)
87 {
88  uint64_t nsec;
89  uint64_t sec = 0;
90  const uint64_t freq_hz = _odp_timer_glob.freq_hz;
91 
92  if (ticks >= freq_hz) {
93  sec = ticks / freq_hz;
94  ticks = ticks - sec * freq_hz;
95  }
96 
97  nsec = (ODP_TIME_SEC_IN_NS * ticks) / freq_hz;
98 
99  return (sec * ODP_TIME_SEC_IN_NS) + nsec;
100 }
101 
102 _ODP_INLINE uint64_t odp_timer_ns_to_tick(odp_timer_pool_t tp ODP_UNUSED, uint64_t ns)
103 {
104  uint64_t ticks;
105  uint64_t sec = 0;
106  const uint64_t freq_hz = _odp_timer_glob.freq_hz;
107 
108  if (ns >= ODP_TIME_SEC_IN_NS) {
109  sec = ns / ODP_TIME_SEC_IN_NS;
110  ns = ns - sec * ODP_TIME_SEC_IN_NS;
111  }
112 
113  ticks = sec * freq_hz;
114  ticks += (ns * freq_hz) / ODP_TIME_SEC_IN_NS;
115 
116  return ticks;
117 }
