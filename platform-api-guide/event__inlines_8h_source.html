69_ODP_INLINE int odp_event_type_multi(const odp_event_t event[], int num,
70 odp_event_type_t *type_out)
71{
72 int i;
73 odp_event_type_t type = __odp_event_type_get(event[0]);
74
75 for (i = 1; i < num; i++) {
76 if (__odp_event_type_get(event[i]) != type)
77 break;
78 }
79
80 *type_out = type;
81
82 return i;
83}

100_ODP_INLINE void *odp_event_user_area(odp_event_t event)
101{
102 const odp_event_type_t type = __odp_event_type_get(event);
103
104 switch (type) {
105 case ODP_EVENT_BUFFER:
106 case ODP_EVENT_ML_COMPL:
107 case ODP_EVENT_DMA_COMPL:
108 return _odp_buffer_get((odp_buffer_t)event, void *, uarea_addr);
109 case ODP_EVENT_PACKET:
110 return _odp_pkt_get((odp_packet_t)event, void *, user_area);
111 case ODP_EVENT_VECTOR:
112 case ODP_EVENT_PACKET_VECTOR:
113 return _odp_event_vect_get(event, void *, uarea_addr);
114 case ODP_EVENT_TIMEOUT:
115 return _odp_timeout_hdr_field((odp_timeout_t)event, void *, uarea_addr);
116 default:
117 return NULL;
118 }
119}

121_ODP_INLINE void *odp_event_user_area_and_flag(odp_event_t event, int *flag)
122{
123 const odp_event_type_t type = __odp_event_type_get(event);
124
125 _ODP_ASSERT(flag != NULL);
126
127 switch (type) {
128 case ODP_EVENT_BUFFER:
129 case ODP_EVENT_DMA_COMPL:
130 case ODP_EVENT_ML_COMPL:
131 *flag = -1;
132 return _odp_buffer_get((odp_buffer_t)event, void *, uarea_addr);
133 case ODP_EVENT_PACKET:
134 {
135 _odp_packet_flags_t pkt_flags;
136 odp_packet_t pkt = (odp_packet_t)event;
137
138 pkt_flags.all_flags = _odp_pkt_get(pkt, uint32_t, flags);
139 *flag = pkt_flags.user_flag;
140
141 return _odp_pkt_get(pkt, void *, user_area);
142 }
143 case ODP_EVENT_VECTOR:
144 case ODP_EVENT_PACKET_VECTOR:
145 {
146 _odp_event_vector_flags_t v_flags;
147
148 v_flags.all_flags = _odp_event_vect_get(event, uint32_t, flags);
149 *flag = v_flags.user_flag;
150
151 return _odp_event_vect_get(event, void *, uarea_addr);
152 }
153 case ODP_EVENT_TIMEOUT:
154 *flag = -1;
155 return _odp_timeout_hdr_field((odp_timeout_t)event, void *, uarea_addr);
156 default:
157 *flag = -1;
158 return NULL;
159 }
160}

162_ODP_INLINE void odp_event_user_flag_set(odp_event_t event, int val)
163{
164 const odp_event_type_t type = __odp_event_type_get(event);
165
166 switch (type) {
167 case ODP_EVENT_PACKET:
168 {
169 odp_packet_t pkt = (odp_packet_t)event;
170 _odp_packet_flags_t *flags = _odp_pkt_get_ptr(pkt, _odp_packet_flags_t, flags);
171
172 flags->user_flag = !!val;
173 return;
174 }
175 case ODP_EVENT_VECTOR:
176 case ODP_EVENT_PACKET_VECTOR:
177 {
178 _odp_event_vector_flags_t *flags =
179 _odp_event_vect_get_ptr(event, _odp_event_vector_flags_t, flags);
180
181 flags->user_flag = !!val;
182 return;
183 }
184 default:
185 /* Nothing to do */
186 return;
187 }
188}

205_ODP_INLINE void odp_event_types_multi(const odp_event_t event[], odp_event_type_t type[],
