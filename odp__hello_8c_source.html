58 if (odp_init_global(&inst, NULL, NULL)) {

64 printf("Local init failed.\n");
65 return -1;
66 }
67

69 printf("Hello world from CPU %i!\n", odp_cpu_id());
70 odp_time_wait_ns(ODP_TIME_SEC_IN_NS);
