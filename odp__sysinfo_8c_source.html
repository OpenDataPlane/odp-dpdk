434static void print_cipher_capa(odp_cipher_alg_t cipher)
435{
436 int caps = odp_crypto_cipher_capability(cipher, NULL, 0);
437 int rc, i;
438
439 if (caps <= 0)
440 return;
441
442 odp_crypto_cipher_capability_t capa[caps];
443
444 rc = odp_crypto_cipher_capability(cipher, capa, caps);
445 if (rc < 0)
446 return;
447
448 printf(" %s:\n", cipher_alg_name(cipher));
449 for (i = 0; i < rc; i++)
450 printf(" key %d iv %d\n",
451 capa[i].key_len, capa[i].iv_len);
452}

454static void print_auth_capa(odp_auth_alg_t auth)
455{
456 int caps = odp_crypto_auth_capability(auth, NULL, 0);
457 int rc, i;
458
459 if (caps <= 0)
460 return;
461
462 odp_crypto_auth_capability_t capa[caps];
463
464 rc = odp_crypto_auth_capability(auth, capa, caps);
465 if (rc < 0)
466 return;
467
468 printf(" %s:\n", auth_alg_name(auth));
469 for (i = 0; i < rc; i++) {
470 printf(" digest %d", capa[i].digest_len);
471 if (capa[i].key_len != 0)
472 printf(" key %d", capa[i].key_len);
473 if (capa[i].iv_len != 0)
474 printf(" iv %d", capa[i].iv_len);
475 if (capa[i].aad_len.max != 0)
476 printf(" aad %d, %d, %d",
477 capa[i].aad_len.min, capa[i].aad_len.max,
478 capa[i].aad_len.inc);
479 printf("\n");
480 }
481}

529static int pktio_capability(appl_args_t *appl_args)
530{
531 odp_pool_param_t pool_param;
532 odp_pool_t pool;
533 int ret = 0;
534
535 odp_pool_param_init(&pool_param);
536
537 pool_param.type = ODP_POOL_PACKET;
538 pool_param.pkt.num = 128;
539
540 pool = odp_pool_create("pktio_pool", &pool_param);
541 if (pool == ODP_POOL_INVALID) {
542 ODPH_ERR("Creating packet pool failed\n");
543 return -1;
544 }
545
546 for (int i = 0; i < appl_args->num_pktio; i++) {
547 odp_pktio_param_t param;
548 odp_pktio_t pktio;
549
550 odp_pktio_param_init(&param);
551
552 param.in_mode = ODP_PKTIN_MODE_SCHED;
553 param.out_mode = ODP_PKTOUT_MODE_DIRECT;
554
555 pktio = odp_pktio_open(appl_args->pktio[i].name, pool, &param);
556 if (pktio == ODP_PKTIO_INVALID) {
557 ODPH_ERR("Opening pktio %s failed\n", appl_args->pktio[i].name);
558 ret = -1;
559 break;
560 }
561
562 if (odp_pktio_capability(pktio, &appl_args->pktio[i].capa)) {
563 ODPH_ERR("Reading pktio %s capa failed\n", appl_args->pktio[i].name);
564 ret = -1;
565 }
566
567 if (odp_proto_stats_capability(pktio, &appl_args->pktio[i].proto_stats_capa)) {
568 ODPH_ERR("Reading pktio %s proto stats capa failed\n",
569 appl_args->pktio[i].name);
570 ret = -1;
571 }
572
573 if (odp_pktio_close(pktio)) {
574 ODPH_ERR("Closing pktio %s failed\n", appl_args->pktio[i].name);
575 ret = -1;
576 }
577
578 if (ret)
579 break;
580 }
581

685 odp_timer_pool_t pool;

688 odp_timer_pool_info_t *info = &appl_args->timer.pool_info[appl_args->timer.num];

691 if (ret && i == ODP_CLOCK_DEFAULT) {
692 ODPH_ERR("odp_timer_capability() failed for default clock source: %d\n",
693 ret);
694 return -1;
695 }
696 if (ret == -1)
697 continue;
698 if (ret < -1) {
699 ODPH_ERR("odp_timer_capability() for clock source %d failed: %d\n", i, ret);
700 return -1;
701 }
702
703 odp_timer_pool_param_init(&params);
704 params.clk_src = i;
705 params.res_ns = capa->max_res.res_ns;
706 params.min_tmo = capa->max_res.min_tmo;
707 params.max_tmo = capa->max_res.max_tmo;
708 params.num_timers = 1;
709
710 pool = odp_timer_pool_create("timer_pool", &params);
711 if (pool == ODP_TIMER_POOL_INVALID) {
712 ODPH_ERR("odp_timer_pool_create() failed for clock source: %d\n", i);
713 return -1;
714 }
715
716 if (odp_timer_pool_start_multi(&pool, 1) != 1) {
717 ODPH_ERR("odp_timer_pool_start_multi() failed for clock source: %d\n", i);
718 return -1;
719 }
720
721 ret = odp_timer_pool_info(pool, info);
722 if (ret) {
723 ODPH_ERR("odp_timer_pool_info() for clock source %d failed: %d\n", i, ret);
724 return -1;
725 }
726
727 odp_timer_pool_destroy(pool);
728
729 appl_args->timer.num++;
730 }
731 return 0;

803static void parse_interfaces(appl_args_t *config, const char *optarg)
804{
805 char *tmp_str = strdup(optarg), *tmp;
806
807 if (tmp_str == NULL)
808 return;
809
810 tmp = strtok(tmp_str, ",");
811
812 while (tmp && config->num_pktio < MAX_IFACES) {
813 if (strlen(tmp) + 1 > MAX_NAME_LEN) {
814 ODPH_ERR("Unable to store interface name (MAX_NAME_LEN=%d)\n",
815 MAX_NAME_LEN);
816 exit(EXIT_FAILURE);
817 }
818 odph_strcpy(config->pktio[config->num_pktio].name, tmp, MAX_NAME_LEN);
819
820 config->num_pktio++;
821
822 tmp = strtok(NULL, ",");
823 }
824
825 free(tmp_str);
826}

829{
830 int opt;
831 static const struct option longopts[] = {
832 {"interfaces", required_argument, NULL, 'i'},
833 {"help", no_argument, NULL, 'h'},
834 {NULL, 0, NULL, 0}
835 };
836 static const char *shortopts = "i:h";
837
838 while (1) {
839 opt = getopt_long(argc, argv, shortopts, longopts, NULL);
840
841 if (opt == -1)
842 break; /* No more options */
843
844 switch (opt) {
845 case 'i':
846 parse_interfaces(appl_args, optarg);

848 case 'h':
849 usage();
850 exit(EXIT_SUCCESS);

855 }
