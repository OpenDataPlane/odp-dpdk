51 static void print_log(test_globals_t *gbls)
52 {
53  uint32_t err_num;
54  int i, j, k, pad;
55  char line[LOG_LINE_SIZE];
56 
57  memset(line, '-', LOG_LINE_SIZE - 1);
58  line[LOG_LINE_SIZE - 1] = 0;
59  for (i = 1; i <= gbls->thread_num; i++) {
60  printf("\n==== history of %d buffer, time,ns (thread) ====\n%s\n",
61  i, line);
62 
63  /* print log for buffer */
64  k = 0;
65  for (j = 0; j < gbls->log_enries_num; j++)
66  if (gbls->log[j].id == i) {
67  printf("%10" PRIu64 " (%-3d)",
68  odp_time_to_ns(gbls->log[j].time),
69  gbls->log[j].thr);
70 
71  if (!(++k % LOG_BASE))
72  printf(" |\n");
73  else
74  printf(" =>");
75  }
76 
77  if ((k % LOG_BASE)) {
78  pad = (LOG_BASE - k % LOG_BASE) * LOG_ENTRY_SIZE - 4;
79  printf(" end%*c\n%s\n", pad, '|', line);
80  } else {
81  printf("%s\n", line);
82  }
83  }
84 
85  printf("\n\n");
86 
87  err_num = odp_atomic_load_u32(&gbls->err_counter);
88  if (err_num)
89  printf("Number of errors: %u\n", err_num);
90 }

92 static void generate_next_queue(test_globals_t *gbls, odp_queue_t *queue,
93  unsigned int id)
94 {
95  int thr;
96  uint8_t rand_u8;
97  char queue_name[sizeof(QUEUE_NAME_PREFIX) + 2];
98  unsigned int rand_id = 1;
99 
100  thr = odp_thread_id();
101 
102  /* generate next random id */
103  if (gbls->thread_num > 1) {
104  do {
105  odp_random_data(&rand_u8, 1, ODP_RANDOM_BASIC);
106  rand_id = rand_u8 % gbls->thread_num + 1;
107  } while (rand_id == id);
108  }
109 
110  sprintf(queue_name, QUEUE_NAME_PREFIX "%d", rand_id);
111  *queue = odp_queue_lookup(queue_name);
112 
113  if (ODP_QUEUE_INVALID == *queue)
114  ODPH_ABORT("Cannot lookup thread queue \"%s\", thread %d\n",
115  queue_name, thr);
116 }

118 static void test_global_timestamps(test_globals_t *gbls,
119  odp_queue_t queue, unsigned int id)
120 {
121  int thr;
122  int log_entry;
123  odp_event_t ev;
124  odp_time_t time;
125  odp_buffer_t buf;
126  odp_queue_t queue_next;
127  timestamp_event_t *timestamp_ev;
128 
129  thr = odp_thread_id();
130  while (odp_atomic_load_u32(&gbls->iteration_counter) <
131  gbls->iteration_num) {
132  ev = odp_queue_deq(queue);
133 
134  if (ev == ODP_EVENT_INVALID)
135  continue;
136 
137  buf = odp_buffer_from_event(ev);
138  timestamp_ev = (timestamp_event_t *)odp_buffer_addr(buf);
139 
140  time = odp_time_global();
141  if (odp_time_cmp(time, timestamp_ev->timestamp) < 0) {
142  ODPH_ERR("timestamp is less than previous time_prev=%"
143  PRIu64 "ns, time_next=%"
144  PRIu64 "ns, thread %d\n",
145  odp_time_to_ns(timestamp_ev->timestamp),
146  odp_time_to_ns(time), thr);
147  odp_atomic_inc_u32(&gbls->err_counter);
148  }
149 
150  /* update the log */
151  log_entry = odp_atomic_fetch_inc_u32(&gbls->log_counter);
152  gbls->log[log_entry].time = timestamp_ev->timestamp;
153  gbls->log[log_entry].id = timestamp_ev->id;
154  gbls->log[log_entry].thr = thr;
155 
156  /* assign new current time and send */
157  generate_next_queue(gbls, &queue_next, id);
158  timestamp_ev->timestamp = time;
159  if (odp_queue_enq(queue_next, ev))
160  ODPH_ABORT("Cannot enqueue event %" PRIu64 " on queue "
161  "%" PRIu64 ", thread %d\n",
162  odp_event_to_u64(ev),
163  odp_queue_to_u64(queue_next), thr);
164 
165  odp_atomic_inc_u32(&gbls->iteration_counter);
166  }
167 }

176 static int run_thread(void *ptr)
177 {
178  int thr;
179  uint32_t id;
180  odp_event_t ev;
181  odp_buffer_t buf;
182  test_globals_t *gbls;
183  odp_pool_t buffer_pool;
184  odp_queue_t queue, queue_next;
185  timestamp_event_t *timestamp_ev;
186  char queue_name[sizeof(QUEUE_NAME_PREFIX) + 2];
187 
188  gbls = ptr;
189  thr = odp_thread_id();
190  printf("Thread %i starts on cpu %i\n", thr, odp_cpu_id());
191 
192  /*
193  * Allocate own queue for receiving timestamps.
194  * Own queue is needed to guarantee that next thread for receiving
195  * buffer is not the same thread.
196  */
197  id = odp_atomic_fetch_inc_u32(&gbls->id_counter);
198  sprintf(queue_name, QUEUE_NAME_PREFIX "%d", id);
199  queue = odp_queue_create(queue_name, NULL);
200  if (queue == ODP_QUEUE_INVALID)
201  ODPH_ABORT("Cannot create thread queue, thread %d", thr);
202 
203  /* allocate buffer for timestamp */
204  buffer_pool = odp_pool_lookup("time buffers pool");
205  if (buffer_pool == ODP_POOL_INVALID)
206  ODPH_ABORT("Buffer pool was not found, thread %d\n", thr);
207 
208  buf = odp_buffer_alloc(buffer_pool);
209  if (buf == ODP_BUFFER_INVALID)
210  ODPH_ABORT("Buffer was not allocated, thread %d\n", thr);
211 
212  /* wait all threads allocated their queues */
213  odp_barrier_wait(&gbls->start_barrier);
214 
215  /* enqueue global timestamp to some queue of some other thread */
216  generate_next_queue(gbls, &queue_next, id);
217 
218  /* save global timestamp and id for tracing */
219  ev = odp_buffer_to_event(buf);
220  timestamp_ev = (timestamp_event_t *)odp_buffer_addr(buf);
221  timestamp_ev->id = id;
222  timestamp_ev->timestamp = odp_time_global();
223  if (odp_queue_enq(queue_next, ev))
224  ODPH_ABORT("Cannot enqueue timestamp event %" PRIu64 " on "
225  "queue %" PRIu64 ", thread %d", odp_event_to_u64(ev),
226  odp_queue_to_u64(queue_next), thr);
227 
228  test_global_timestamps(gbls, queue, id);
229 
230  /* wait all threads are finished their jobs */
231  odp_barrier_wait(&gbls->end_barrier);
232 
233  /* free all events on the allocated queue */
234  while (1) {
235  ev = odp_queue_deq(queue);
236  if (ev == ODP_EVENT_INVALID)
237  break;
238 
239  buf = odp_buffer_from_event(ev);
240  odp_buffer_free(buf);
241  }
242 
243  /* free allocated queue */
244  if (odp_queue_destroy(queue))
245  ODPH_ABORT("Cannot destroy queue %" PRIu64 "",
246  odp_queue_to_u64(queue));
247 
248  printf("Thread %i exits\n", thr);
249  fflush(NULL);
250  return 0;
251 }

253 int main(int argc, char *argv[])
254 {
255  int err = 0;
256  odp_pool_t pool = ODP_POOL_INVALID;
257  int num_workers;
258  test_globals_t *gbls;
259  odp_cpumask_t cpumask;
260  odp_pool_capability_t pool_capa;
261  odp_pool_param_t pool_param;
262  odp_shm_t shm_glbls = ODP_SHM_INVALID;
263  odp_shm_t shm_log = ODP_SHM_INVALID;
264  int log_size, log_enries_num;
265  odph_helper_options_t helper_options;
266  odph_thread_t thread_tbl[MAX_WORKERS];
267  odp_instance_t instance;
268  odp_init_t init_param;
269  odph_thread_common_param_t thr_common;
270  odph_thread_param_t thr_param;
271 
272  printf("\nODP global time test starts\n");
273 
274  /* Let helper collect its own arguments (e.g. --odph_proc) */
275  argc = odph_parse_options(argc, argv);
276  if (odph_options(&helper_options)) {
277  ODPH_ERR("Error: reading ODP helper options failed.\n");
278  exit(EXIT_FAILURE);
279  }
280 
281  odp_init_param_init(&init_param);
282  init_param.mem_model = helper_options.mem_model;
283 
284  if (odp_init_global(&instance, &init_param, NULL)) {
285  err = 1;
286  ODPH_ERR("ODP global init failed.\n");
287  goto end;
288  }
289 
290  /* Init this thread. */
291  if (odp_init_local(instance, ODP_THREAD_CONTROL)) {
292  err = 1;
293  ODPH_ERR("ODP local init failed.\n");
294  goto err_global;
295  }
296 
297  num_workers = MAX_WORKERS;
298  num_workers = odp_cpumask_default_worker(&cpumask, num_workers);
299 
300  shm_glbls = odp_shm_reserve("test_globals", sizeof(test_globals_t),
301  ODP_CACHE_LINE_SIZE, 0);
302  if (ODP_SHM_INVALID == shm_glbls) {
303  err = 1;
304  ODPH_ERR("Error: shared mem reserve failed.\n");
305  goto err;
306  }
307 
308  log_enries_num = num_workers * (ITERATION_NUM + num_workers);
309  log_size = sizeof(log_entry_t) * log_enries_num;
310  shm_log = odp_shm_reserve("test_log", log_size, ODP_CACHE_LINE_SIZE, 0);
311  if (ODP_SHM_INVALID == shm_log) {
312  err = 1;
313  ODPH_ERR("Error: shared mem reserve failed.\n");
314  goto err;
315  }
316 
317  gbls = odp_shm_addr(shm_glbls);
318  gbls->thread_num = num_workers;
319  gbls->iteration_num = ITERATION_NUM;
320  odp_atomic_store_u32(&gbls->iteration_counter, 0);
321  odp_atomic_store_u32(&gbls->id_counter, 1);
322  odp_atomic_store_u32(&gbls->log_counter, 0);
323  odp_atomic_store_u32(&gbls->err_counter, 0);
324  gbls->log_enries_num = log_enries_num;
325  gbls->log = odp_shm_addr(shm_log);
326  odp_barrier_init(&gbls->start_barrier, num_workers);
327  odp_barrier_init(&gbls->end_barrier, num_workers);
328  memset(gbls->log, 0, log_size);
329 
330  if (odp_pool_capability(&pool_capa)) {
331  err = 1;
332  ODPH_ERR("Error: pool capability failed.\n");
333  goto err;
334  }
335 
336  odp_pool_param_init(&pool_param);
337 
338  pool_param.buf.size = sizeof(timestamp_event_t);
339  pool_param.buf.num = MAX_NUM_BUF;
340  pool_param.type = ODP_POOL_BUFFER;
341 
342  if (pool_capa.buf.max_num && MAX_NUM_BUF > pool_capa.buf.max_num)
343  pool_param.buf.num = pool_capa.buf.max_num;
344 
345  pool = odp_pool_create("time buffers pool", &pool_param);
346  if (pool == ODP_POOL_INVALID) {
347  err = 1;
348  ODPH_ERR("Pool create failed.\n");
349  goto err;
350  }
351 
352  odph_thread_common_param_init(&thr_common);
353  odph_thread_param_init(&thr_param);
354 
355  thr_param.start = run_thread;
356  thr_param.arg = gbls;
357  thr_param.thr_type = ODP_THREAD_WORKER;
358 
359  thr_common.instance = instance;
360  thr_common.cpumask = &cpumask;
361  thr_common.share_param = 1;
362 
363  /* Create and launch worker threads */
364  odph_thread_create(thread_tbl, &thr_common, &thr_param, num_workers);
365 
366  /* Wait for worker threads to exit */
367  odph_thread_join(thread_tbl, num_workers);
368 
369  print_log(gbls);
370 
371 err:
372  if (pool != ODP_POOL_INVALID)
373  if (odp_pool_destroy(pool))
374  err = 1;
375 
376  if (shm_log != ODP_SHM_INVALID)
377  if (odp_shm_free(shm_log))
378  err = 1;
379 
380  if (shm_glbls != ODP_SHM_INVALID)
381  if (odp_shm_free(shm_glbls))
382  err = 1;
383 
384  if (odp_term_local())
385  err = 1;
386 err_global:
387  if (odp_term_global(instance))
388  err = 1;
389 end:
390  if (err) {
391  ODPH_ERR("Err: ODP global time test failed\n\n");
392  return -1;
393  }
394 
395  printf("ODP global time test complete\n\n");
396  return 0;
397 }
